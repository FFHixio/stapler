001package org.kohsuke.stapler.json;
002
003import com.gargoylesoftware.htmlunit.HttpMethod;
004import com.gargoylesoftware.htmlunit.WebClient;
005import com.gargoylesoftware.htmlunit.WebRequestSettings;
006import com.gargoylesoftware.htmlunit.WebResponse;
007import java.net.URL;
008import static junit.framework.TestCase.assertEquals;
009import org.kohsuke.stapler.test.JettyTestCase;
010
011public class JsonBodyTest extends JettyTestCase {
012
013    public void testSmokes() throws Exception {
014        WebRequestSettings req = new WebRequestSettings(new URL(url, "double"), HttpMethod.POST);
015        req.setAdditionalHeader("Content-Type", "application/json");
016        req.setRequestBody("{\"x\":10,\"y\":5}");
017        WebResponse response = new WebClient().getPage(req).getWebResponse();
018        assertEquals("application/json", response.getContentType());
019        assertEquals("{\"x\":20,\"y\":10}", response.getContentAsString());
020        // and then with compression:
021        req.setAdditionalHeader("Accept-Encoding", "gzip");
022        response = new WebClient().getPage(req).getWebResponse();
023        assertEquals("application/json", response.getContentType());
024        assertEquals("{\"x\":20,\"y\":10}", response.getContentAsString());
025    }
026
027    @JsonResponse
028    public Point doDouble(@JsonBody Point p) {
029        Point pt = new Point();
030        pt.x = p.x * 2;
031        pt.y = p.y * 2;
032        return pt;
033    }
034
035    public static class Point {
036        public int x, y;
037    }
