024package org.kohsuke.stapler;
025
026import java.util.StringTokenizer;
027import java.io.ByteArrayOutputStream;
028import java.io.UnsupportedEncodingException;
029
030/**
031 * Tokenized path portion of the URL.
032 *
033 * For example "foo/bar/zot" is treated as ["foo","bar","zot"]
034 *
035 * @author Kohsuke Kawaguchi
036 */
037public final class TokenList {
038    /**
039     * URL-decoded tokens.
040     */
041    public final String[] tokens;
042    /**
043     * Like {@link #tokens} but before decoding.
044     */
045    public final String[] rawTokens;
046    /**
047     * Index of the next token.
048     */
049    public int idx;
050
051    /**
052     * If the request URL ends with the path separator.
053     */
054    public final boolean endsWithSlash;
055
056    TokenList(String url) {
057        // to avoid a directory traversal vulnerability in Windows, treat '\\' as a path separator just like '/'
058        StringTokenizer tknzr = new StringTokenizer(url,"/\\");
059        final int tokenCount = tknzr.countTokens();
060        tokens = new String[tokenCount];
061        rawTokens = new String[tokenCount];
062        for(int i=0; tknzr.hasMoreTokens(); i++) {
063            rawTokens[i] = tknzr.nextToken();
064            tokens[i] = decode(rawTokens[i]);
065            if (tokens[i].equals(".."))
066                throw new IllegalArgumentException(url);
067        }
068        endsWithSlash = url.endsWith("/") || url.endsWith("\\");
069    }
070
071    public boolean hasMore() {
072        return tokens.length!=idx;
073    }
074
075    public String peek() {
076        if(hasMore())
077            return tokens[idx];
078        else
079            return null;
080    }
081
082    public String next() {
083        return tokens[idx++];
084    }
085    public String prev() {
086        return tokens[--idx];
087    }
088    public int nextAsInt() throws NumberFormatException {
089        long asLongValue = nextAsLong();
090        if (asLongValue < Integer.MIN_VALUE) {
091            throw new NumberFormatException(String.format("Token '%d' cannot be interpreted as an integer as its value is less than %d.", asLongValue, Integer.MIN_VALUE));
092        } else if (asLongValue > Integer.MAX_VALUE) {
093            throw new NumberFormatException(String.format("Token '%d' cannot be interpreted as an integer as its value is greater than %d.", asLongValue, Integer.MAX_VALUE));
094        }
095
096        return (int) asLongValue;
097    }
098    public long nextAsLong() throws NumberFormatException {
099        String p = peek();
100        if(p == null) {
101            throw new NumberFormatException();  // no more token
102        }
103        long asLongValue = Long.parseLong(p);
104        idx++;
105        return asLongValue;
106    }
107
108    public int length() {
109        return tokens.length;
110    }
111
112    public String get(int i) {
113        return tokens[i];
114    }
115
116    public int countRemainingTokens() {
117        return length()-idx;
118    }
119
120
121    public String toString() {
122        StringBuilder buf = new StringBuilder();
123        for( int i=0; i<tokens.length; i++) {
124            if(buf.length()>0)  buf.append('/');
125            if(i==idx)  buf.append('!');
126            buf.append(tokens[i]);
127        }
128        return buf.toString();
129    }
130
131    private String assembleRestOfPath(String[] tokens) {
132        StringBuilder buf = new StringBuilder();
133        for( int i=idx; i<length(); i++ ) {
134            buf.append('/');
135            buf.append(tokens[i]);
136        }
137        return buf.toString();
138    }
139
140    public String assembleRestOfPath() {
141        return assembleRestOfPath(tokens);
142    }
143
144    public String assembleOriginalRestOfPath() {
145        return assembleRestOfPath(rawTokens);
146    }
147
148    public static String decode(String s) {
149        int i = s.indexOf('%');
150        if (i < 0) return s;
151
152        try {
153            // to properly handle non-ASCII characters, decoded bytes need to be stored and translated in bulk.
154            // this complex set up is necessary for us to work gracefully if 's' already contains decoded non-ASCII chars.
155            ByteArrayOutputStream baos = new ByteArrayOutputStream();
156
157            StringBuilder buf = new StringBuilder(s.substring(0, i));
158            char c, upper, lower;
159            for (int m = s.length(); i < m; i++) {
160                c = s.charAt(i);
161                if (c == '%') {
162                    try {
163                        upper = s.charAt(++i);
164                        lower = s.charAt(++i);
165                        baos.write(fromHex(upper) * 16 + fromHex(lower));
166                    } catch (IndexOutOfBoundsException ignore) {
167                        // malformed %HH.
168                    }
169                } else {
170                    if (baos.size()>0) {
171                        buf.append(new String(baos.toByteArray(),"UTF-8"));
172                        baos.reset();
173                    }
174                    buf.append(c);
175                }
176            }
177            if (baos.size()>0)
178                buf.append(new String(baos.toByteArray(),"UTF-8"));
179            return buf.toString();
180        } catch (UnsupportedEncodingException e) {
181            throw new AssertionError(e); // UTF-8 is mandatory encoding
182        }
183    }
184
185    private static int fromHex(char upper) {
186        return ((upper & 0xF) + ((upper & 0x40) != 0 ? 9 : 0));
187    }
188}
