001package org.kohsuke.stapler.jsr269;
002
003import org.kohsuke.MetaInfServices;
004import org.kohsuke.stapler.DataBoundConstructor;
005
006import javax.annotation.processing.Processor;
007import javax.annotation.processing.RoundEnvironment;
008import javax.annotation.processing.SupportedAnnotationTypes;
009import javax.lang.model.SourceVersion;
010import javax.lang.model.element.Element;
011import javax.lang.model.element.ElementKind;
012import javax.lang.model.element.ExecutableElement;
013import javax.lang.model.element.TypeElement;
014import javax.lang.model.element.VariableElement;
015import javax.lang.model.util.ElementScanner6;
016import java.io.IOException;
017import java.util.HashSet;
018import java.util.Properties;
019import java.util.Set;
020import javax.lang.model.element.Modifier;
021import javax.tools.Diagnostic;
022
023/**
024 * @author Kohsuke Kawaguchi
025 */
026@SuppressWarnings({"Since15"})
027@SupportedAnnotationTypes("*")
028@MetaInfServices(Processor.class)
029public class ConstructorProcessor extends AbstractProcessorImpl {
030   /* private */ final static String MESSAGE = "Only one annotated constructor (@DataBoundConstructor annotation or @stapler-constructor javadoc) is permitted per class";
031
032    @Override
033    public boolean process(Set<? extends TypeElement> annotations, RoundEnvironment roundEnv) {
034        try {
035            ElementScanner6<Void, Void> scanner = new ElementScanner6<Void, Void>() {
036                Set<Element> enclosingElementsWritten = new HashSet<>();
037                boolean messagePrinted;
038
039                @Override
040                public Void visitExecutable(ExecutableElement e, Void aVoid) {
041                    if(e.getAnnotation(DataBoundConstructor.class)!=null) {
042                        writeOrAddOnlyOneMessage(e);
043                    } else {
044                        String javadoc = getJavadoc(e);
045                        if(javadoc!=null && javadoc.contains("@stapler-constructor")) {
046                            writeOrAddOnlyOneMessage(e);
047                        }
048                    }
049
050                    return super.visitExecutable(e, aVoid);
051                }
052
053                @Override
054                public Void visitUnknown(Element e, Void aVoid) {
055                    return DEFAULT_VALUE;
056                }
057
058                private void writeOrAddOnlyOneMessage(ExecutableElement e) {
059                    if (enclosingElementsWritten.add(e.getEnclosingElement())) {
060                        write(e);
061                    } else if (!messagePrinted){
062                        processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, MESSAGE, e);
063                        messagePrinted = true;
064                    }
065                }
066            };
067
068            for (Element e : roundEnv.getRootElements()) {
069                if (e.getKind() == ElementKind.PACKAGE) { // JENKINS-11739
070                    continue;
071                }
072                scanner.scan(e, null);
073            }
074
075            return false;
076        } catch (RuntimeException e) {
077            // javac sucks at reporting errors in annotation processors
078            e.printStackTrace();
079            throw e;
080        } catch (Error e) {
081            e.printStackTrace();
082            throw e;
083        }
084    }
085
086    @Override
087    public SourceVersion getSupportedSourceVersion() {
088        return SourceVersion.latest();
089    }
090
091    private void write(ExecutableElement c) {
092        if (!c.getModifiers().contains(Modifier.PUBLIC)) {
093            processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, "@DataBoundConstructor must be applied to a public constructor", c);
094            return;
095        }
096        if (c.getEnclosingElement().getModifiers().contains(Modifier.ABSTRACT)) {
097            processingEnv.getMessager().printMessage(Diagnostic.Kind.ERROR, "@DataBoundConstructor may not be used on an abstract class (only on concrete subclasses)", c);
098            return;
099        }
100        try {
101            StringBuilder buf = new StringBuilder();
102            for( VariableElement p : c.getParameters() ) {
103                if(buf.length()>0)  buf.append(',');
104                buf.append(p.getSimpleName());
105            }
106
107            TypeElement t = (TypeElement) c.getEnclosingElement();
108            String name = t.getQualifiedName().toString().replace('.', '/') + ".stapler";
109            notice("Generating " + name, c);
110
111            Properties p = new Properties();
112            p.put("constructor",buf.toString());
113            writePropertyFile(p, name);
114        } catch (IOException x) {
115            error(x);
116        }
117    }
118}
