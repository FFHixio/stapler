024package org.kohsuke.stapler;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027
028import javax.servlet.ServletException;
029import javax.servlet.RequestDispatcher;
030import java.io.IOException;
031import java.util.HashMap;
032import java.util.Map;
033import java.util.Map.Entry;
034
035/**
036 * {@link HttpResponse} that forwards to a {@link RequestDispatcher}, such as a view.
037 * Extends from {@link RuntimeException} so that you can throw it.
038 *
039 * @author Kohsuke Kawaguchi
040 */
041public class ForwardToView extends RuntimeException implements HttpResponse {
042    @SuppressFBWarnings(value = "SE_BAD_FIELD", justification = "As an HttpResponse, this is not actually serialized.")
043    private final DispatcherFactory factory;
044    private final Map<String,Object> attributes = new HashMap<String, Object>();
045
046    private interface DispatcherFactory {
047        RequestDispatcher get(StaplerRequest req) throws IOException;
048    }
049
050    public ForwardToView(final RequestDispatcher dispatcher) {
051        this.factory = new DispatcherFactory() {
052            public RequestDispatcher get(StaplerRequest req) {
053                return dispatcher;
054            }
055        };
056    }
057
058    public ForwardToView(final Object it, final String view) {
059        this.factory = new DispatcherFactory() {
060            public RequestDispatcher get(StaplerRequest req) throws IOException {
061                return req.getView(it,view);
062            }
063        };
064    }
065
066    public ForwardToView(final Class c, final String view) {
067        this.factory = new DispatcherFactory() {
068            public RequestDispatcher get(StaplerRequest req) throws IOException {
069                return req.getView(c,view);
070            }
071        };
072    }
073
074    /**
075     * Forwards to the view with specified attributes exposed as a variable binding.
076     */
077    public ForwardToView with(String varName, Object value) {
078        attributes.put(varName,value);
079        return this;
080    }
081
082    public ForwardToView with(Map<String,?> attributes) {
083        this.attributes.putAll(attributes);
084        return this;
085    }
086
087    /**
088     * Make this forwarding optional. Render nothing if a view doesn't exist.
089     */
090    public ForwardToView optional() {
091        return this;
092    }
093
094    @SuppressFBWarnings(value = "REQUESTDISPATCHER_FILE_DISCLOSURE", justification = "Forwarded to a view to handle correctly.")
095    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
096        for (Entry<String, Object> e : attributes.entrySet())
097            req.setAttribute(e.getKey(),e.getValue());
098        RequestDispatcher rd = factory.get(req);
099        if (rd != null) {
100            rd.forward(req, rsp);
101        }
102    }
103}
