024package org.kohsuke.stapler.framework.adjunct;
025
026import org.kohsuke.stapler.*;
027
028import javax.servlet.ServletException;
029import javax.servlet.ServletContext;
030import static javax.servlet.http.HttpServletResponse.SC_FORBIDDEN;
031import static javax.servlet.http.HttpServletResponse.SC_NOT_FOUND;
032import java.io.IOException;
033import java.net.URL;
034import java.util.concurrent.ConcurrentHashMap;
035import java.util.concurrent.ConcurrentSkipListSet;
036
037/**
038 * This application-scoped object that exposes djuncts to URL.
039 *
040 * <p>
041 * Adjuncts are packaging of JavaScript, CSS, and other static assets in jar files with dependency
042 * information between them. This allows JavaScript libraries and other static assets to be reused
043 * across different projects through Maven/Ivy.
044 *
045 * <p>
046 * To use {@link AdjunctManager} in your application, create one instance, and bind it to URL
047 * (like you do any other objects.) The most typical way of doing this is to define it as a
048 * field in your top-level object.
049 *
050 * <pre>
051 * public class MyApplication {
052 *     public final AdjunctManager adjuncts = new AdjunctManager(context,getClass().getClassLoader(),"/adjuncts");
053 * }
054 * </pre>
055 *
056 * <p>
057 * How you include an adjunct will depend on your template language, but for example in Jelly you do:
058 * <pre>{@code <st:adjunct includes="org.kohsuke.stapler.bootstrap"/>}</pre>
059 *
060 * Or from Groovy you do:
061 * <pre>{@code adjunct "org.kohsuke.stapler.bootstrap"}</pre>
062 *
063 * <p>
064 * ... and this produces a series of <tt>style</tt> and <tt>script</tt> tags that include all the
065 * necessary JavaScript, CSS, and their dependencies.
066 *
067 * <p>
068 * Internally, this class provides caching for {@link Adjunct}s.
069 *
070 * @author Kohsuke Kawaguchi
071 * @see Adjunct
072 */
073public class AdjunctManager {
074    private final ConcurrentHashMap<String, Adjunct> adjuncts = new ConcurrentHashMap<String,Adjunct>();
075
076    /**
077     * Map used as a set to remember which resources can be served.
078     */
079    private final ConcurrentSkipListSet<String> allowedResources = new ConcurrentSkipListSet<>();
080
081    private final ClassLoader classLoader;
082
083    /**
084     * Absolute URL of the {@link AdjunctManager} in the calling application where it is bound to.
085     *
086     * <p>
087     * The path is treated relative from the context path of the application, and it
088     * needs to end without '/'. So it needs to be something like "foo/adjuncts" or
089     * just "adjuncts". Can be e.g. {@code adjuncts/uNiQuEhAsH} to improve caching behavior.
090     */
091    public final String rootURL;
092
093    /**
094     * Hint instructing adjuncts to load a debuggable non-minified version of the script,
095     * as opposed to the production version.
096     *
097     * This is only a hint, and so the semantics of it isn't very well defined. The intention
098     * is to assist JavaScript debugging.
099     */
100    public boolean debug = Boolean.getBoolean(AdjunctManager.class.getName()+".debug");
101
102    public final WebApp webApp;
103    private final long expiration;
104
105    @Deprecated
106    public AdjunctManager(ServletContext context,ClassLoader classLoader, String rootURL) {
107        this(context, classLoader, rootURL, /* one day */24L * 60 * 60 * 1000);
108    }
109
110    /**
111     * @param classLoader
112     *      ClassLoader to load adjuncts from.
113     * @param rootURL
114     *      See {@link #rootURL} for the meaning of this parameter.
115     * @param expiration milliseconds from service time until expiration, for {@link #doDynamic}
116     *                    (as in {@link StaplerResponse#serveFile(StaplerRequest, URL, long)});
117     *                    if {@link #rootURL} is unique per session then this can be very long;
118     *                    otherwise a day might be reasonable
119     */
120    public AdjunctManager(ServletContext context, ClassLoader classLoader, String rootURL, long expiration) {
121        this.classLoader = classLoader;
122        this.rootURL = rootURL;
123        this.webApp = WebApp.get(context);
124        this.expiration = expiration;
125        // register this globally
126        context.setAttribute(KEY,this);
127    }
128
129    public static AdjunctManager get(ServletContext context) {
130        return (AdjunctManager) context.getAttribute(KEY);
131    }
132
133    /**
134     * Obtains the adjunct.
135     *
136     * @return
137     *      always non-null.
138     * @throws IOException
139     *      if failed to locate {@link Adjunct}.
140     */
141    public Adjunct get(String name) throws IOException {
142        Adjunct a = adjuncts.get(name);
143        if(a!=null) return a;   // found it
144
145        synchronized (this) {
146            a = adjuncts.get(name);
147            if(a!=null) return a;   // one more check before we start loading
148            a = new Adjunct(this,name,classLoader);
149            adjuncts.put(name,a);
150            return a;
151        }
152    }
153
154    /**
155     * Serves resources in the class loader.
156     */
157    public void doDynamic(StaplerRequest req, StaplerResponse rsp) throws IOException, ServletException {
158        String path = req.getRestOfPath();
159        if (path.charAt(0)=='/') path = path.substring(1);
160
161        if(!allowedResources.contains(path)) {
162            if(!allowResourceToBeServed(path)) {
163                rsp.sendError(SC_FORBIDDEN);
164                return;
165            }
166            // remember URLs that we can serve. but don't remember error ones, as it might be unbounded
167            allowedResources.add(path);
168        }
169
170        URL res = classLoader.getResource(path);
171        if(res==null) {
172            throw HttpResponses.error(SC_NOT_FOUND,new IllegalArgumentException("No such adjunct found: "+path));
173        } else {
174            long expires = MetaClass.NO_CACHE ? 0 : expiration;
175            rsp.serveFile(req,res,expires);
176        }
177    }
178
179    /**
180     * Controls whether the given resource can be served to browsers.
181     *
182     * <p>
183     * This method can be overridden by the sub classes to change the access control behavior.
184     *
185     * <p>
186     * {@link AdjunctManager} is capable of serving all the resources visible
187     * in the classloader by default. If the resource files need to be kept private,
188     * return false, which causes the request to fail with 401. 
189     *
190     * Otherwise return true, in which case the resource will be served.
191     */
192    protected boolean allowResourceToBeServed(String absolutePath) {
193        // does it have an adjunct directory marker?
194        int idx = absolutePath.lastIndexOf('/');
195        if (idx>0 && classLoader.getResource(absolutePath.substring(0,idx)+"/.adjunct")!=null)
196            return true;
197
198        // backward compatible behaviour
199        return absolutePath.endsWith(".gif")
200            || absolutePath.endsWith(".png")
201            || absolutePath.endsWith(".css")
202            || absolutePath.endsWith(".js");
203    }
204
205    /**
206     * Key in {@link ServletContext} to look up {@link AdjunctManager}.
207     */
208    private static final String KEY = AdjunctManager.class.getName();
209}
