024package org.kohsuke.stapler.framework.io;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027
028import java.io.FileWriter;
029import java.io.Writer;
030import java.io.File;
031import java.io.IOException;
032import java.io.BufferedWriter;
033import java.io.OutputStreamWriter;
034import java.nio.file.Files;
035import java.nio.file.Path;
036import java.nio.file.StandardOpenOption;
037
038/**
039 * Buffered {@link FileWriter} that uses UTF-8.
040 *
041 * <p>
042 * The write operation is atomic when used for overwriting;
043 * it either leaves the original file intact, or it completely rewrites it with new contents.
044 *
045 * @author Kohsuke Kawaguchi
046 */
047public class AtomicFileWriter extends Writer {
048
049    private final Writer core;
050    private final File tmpFile;
051    private final File destFile;
052
053    @SuppressFBWarnings(value = "PATH_TRAVERSAL_IN", justification = "Protected by checks at other layers.")
054    public AtomicFileWriter(File f) throws IOException {
055        tmpFile = File.createTempFile("atomic",null,f.getParentFile());
056        destFile = f;
057        core = new BufferedWriter(new OutputStreamWriter(Files.newOutputStream(
058                tmpFile.toPath(),
059                StandardOpenOption.CREATE,
060                StandardOpenOption.TRUNCATE_EXISTING, // empty file already created by createTempFile
061                StandardOpenOption.WRITE
062        ),"UTF-8"));
063    }
064
065    public void write(int c) throws IOException {
066        core.write(c);
067    }
068
069    public void write(String str, int off, int len) throws IOException {
070        core.write(str,off,len);
071    }
072
073    public void write(char cbuf[], int off, int len) throws IOException {
074        core.write(cbuf,off,len);
075    }
076
077    public void flush() throws IOException {
078        core.flush();
079    }
080
081    public void close() throws IOException {
082        core.close();
083    }
084
085    public void commit() throws IOException {
086        close();
087        Path destFilePath = destFile.toPath();
088        if (Files.exists(destFilePath)) {
089            Files.delete(destFilePath);
090        }
091        Files.move(tmpFile.toPath(), destFilePath);
092    }
093
094    /**
095     * Until the data is committed, this file captures
096     * the written content.
097     */
098    public File getTemporaryFile() {
099        return tmpFile;
100    }
101}
