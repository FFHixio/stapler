100     * uncompress its content during read-access. Do note that the underlying
101     * file is not altered and remains compressed.

103    public LargeText(final File file, Charset charset, boolean completed, boolean transparentGunzip) {
104        this.charset = charset;
105        if (transparentGunzip && GzipAwareSession.isGzipStream(file)) {
106            this.source = new Source() {
107                public Session open() throws IOException {
108                    return new GzipAwareSession(file);
109                }
110    
111                public long length() {
112                    return GzipAwareSession.getGzipStreamSize(file);
113                }
114    
115                public boolean exists() {
116                    return file.exists();
117                }
118            };
119        } else {
120            this.source = new Source() {
121                public Session open() throws IOException {
122                    return new FileSession(file);
123                }
124    
125                public long length() {
126                    return file.length();
127                }
128    
129                public boolean exists() {
130                    return file.exists();
131                }
132            };
133        }
134        this.completed = completed;
135    }

171    /**
172     * Returns {@link Reader} for reading the raw bytes.
173     */
174    public Reader readAll() throws IOException {
175        return new InputStreamReader(new InputStream() {
176            final Session session = source.open();
177            public int read() throws IOException {
178                byte[] buf = new byte[1];
179                int n = session.read(buf);
180                if(n==1)    return buf[0];
181                else        return -1; // EOF
182            }
183
184            public int read(byte[] buf, int off, int len) throws IOException {
185                return session.read(buf,off,len);
186            }
187
188            public void close() throws IOException {
189                session.close();
190            }
191        },charset);
192    }

194    public long writeLogTo(long start, Writer w) throws IOException {
195        return writeLogTo(start, new WriterOutputStream(w, charset));
196    }

198    /**
199     * Writes the tail portion of the file to the {@link OutputStream}.
200     *
201     * @param start
202     *      The byte offset in the input file where the write operation starts.
203     *
204     * @return
205     *      if the file is still being written, this method writes the file
206     *      until the last newline character and returns the offset to start
207     *      the next write operation.
208     * @throws EOFException if the start position is larger than the file size
209     */
210    public long writeLogTo(long start, OutputStream out) throws IOException {
211        CountingOutputStream os = new CountingOutputStream(out);
212
213        try (Session f = source.open()) {
214            f.skip(start);
215
216            if (completed) {
217                // write everything till EOF
218                byte[] buf = new byte[1024];
219                int sz;
220                while ((sz = f.read(buf)) >= 0)
221                    os.write(buf, 0, sz);
222            } else {
223                ByteBuf buf = new ByteBuf(null, f);
224                HeadMark head = new HeadMark(buf);
225                TailMark tail = new TailMark(buf);
226                buf = null;
227
228                int readLines = 0;
229                while (tail.moveToNextLine(f) && readLines++ < MAX_LINES_READ) {
230                    head.moveTo(tail, os);
231                }
232                head.finish(os);
233            }
234        }
235
236        os.flush();
237
238        return os.getByteCount()+start;
239    }

241    /**
242     * Implements the progressive text handling.
243     * This method is used as a "web method" with progressiveText.jelly.
244     */
245    public void doProgressText(StaplerRequest req, StaplerResponse rsp) throws IOException {
246        setContentType(rsp);
247        rsp.setStatus(HttpServletResponse.SC_OK);
248
249        if(!source.exists()) {
250            // file doesn't exist yet
251            rsp.addHeader("X-Text-Size","0");
252            rsp.addHeader("X-More-Data","true");
253            return;
254        }
255
256        long start = 0;
257        String s = req.getParameter("start");
258        if(s!=null)
259            start = Long.parseLong(s);
260
261        if(source.length() < start )
262            start = 0;  // text rolled over
263
264        CharSpool spool = new CharSpool();
265        long r = writeLogTo(start,spool);
266
267        rsp.addHeader("X-Text-Size",String.valueOf(r));
268        if(!completed)
269            rsp.addHeader("X-More-Data","true");
270
271        Writer w = createWriter(req, rsp, r - start);
272        spool.writeTo(new LineEndNormalizingWriter(w));
273        w.close();
274    }
275
276    protected void setContentType(StaplerResponse rsp) {
277        rsp.setContentType("text/plain;charset=UTF-8");
278    }
279
280    protected Writer createWriter(StaplerRequest req, StaplerResponse rsp, long size) throws IOException {
281        // when sending big text, try compression. don't bother if it's small
282        if(size >4096)
283            return rsp.getCompressedWriter(req);
284        else
285            return rsp.getWriter();
286    }

288    /**
289     * Points to a byte in the buffer.
290     */
291    private static class Mark {
292        protected ByteBuf buf;
293        protected int pos;
294
295        public Mark(ByteBuf buf) {
296            this.buf = buf;
297        }
298    }
299
300    /**
301     * Points to the start of the region that's not committed
302     * to the output yet.
303     */
304    private static final class HeadMark extends Mark {
305        public HeadMark(ByteBuf buf) {
306            super(buf);
307        }
308
309        /**
310         * Moves this mark to 'that' mark, and writes the data
311         * in between to {@link OutputStream} if necessary.
312         */
313        void moveTo(Mark that, OutputStream os) throws IOException {
314            while(this.buf!=that.buf) {
315                os.write(buf.buf,0,buf.size);
316                buf = buf.next;
317                pos = 0;
318            }
319
320            this.pos = that.pos;
321        }
322
323        void finish(OutputStream os) throws IOException {
324            os.write(buf.buf,0,pos);
325        }
326    }
327
328    /**
329     * Points to the end of the region.
330     */
331    private static final class TailMark extends Mark {
332        public TailMark(ByteBuf buf) {
333            super(buf);
334        }
335
336        boolean moveToNextLine(Session f) throws IOException {
337            while(true) {
338                while(pos==buf.size) {
339                    if(!buf.isFull()) {
340                        // read until EOF
341                        return false;
342                    } else {
343                        // read into the next buffer
344                        buf = new ByteBuf(buf,f);
345                        pos = 0;
346                    }
347                }
348                byte b = buf.buf[pos++];
349                if(b=='\r' || b=='\n')
350                    return true;
351            }
352        }
353    }

355    /**
356     * Variable length byte buffer implemented as a linked list of fixed length buffer.
357     */
358    private static final class ByteBuf {
359        private final byte[] buf = new byte[1024];
360        private int size = 0;
361        private ByteBuf next;
362
363        public ByteBuf(ByteBuf previous, Session f) throws IOException {
364            if(previous!=null) {
365                assert previous.next==null;
366                previous.next = this;
367            }
368
369            while(!this.isFull()) {
370                int chunk = f.read(buf, size, buf.length - size);
371                if(chunk==-1)
372                    return;
373                size+= chunk;
374            }
375        }
376
377        public boolean isFull() {
378            return buf.length==size;
379        }
380    }

423    /**
424     * {@link Session} implementation over {@link GZIPInputStream}.
425     * <p>
426     * Always use {@link GzipAwareSession#isGzipStream(File)} to check if you
427     * really deal with a GZIPed file before you invoke this class. Otherwise,
428     * {@link GZIPInputStream} might throw an exception.
429     */
430    private static final class GzipAwareSession implements Session {
431        private final GZIPInputStream gz;
432
433        public GzipAwareSession(File file) throws IOException {
434            this.gz = new GZIPInputStream(Files.newInputStream(file.toPath(), StandardOpenOption.READ));
435        }
436
437        @Override
438        public void close() throws IOException {
439            gz.close();
440        }
441
442        @Override
443        public void skip(long start) throws IOException {
444            while (start > 0) {
445                start -= gz.skip(start);
446            }
447        }
448
449        @Override
450        public int read(byte[] buf) throws IOException {
451            return gz.read(buf);
452        }
453
454        @Override
455        public int read(byte[] buf, int offset, int length) throws IOException {
456            return gz.read(buf,offset,length);
457        }
458    
459        /**
460         * Checks the first two bytes of the target file and return true if
461         * they equal the GZIP magic number.
462         * @param file
463         * @return true, if the first two bytes are the GZIP magic number.
464         */
465        public static boolean isGzipStream(File file) {
466            try (InputStream in = Files.newInputStream(file.toPath(), StandardOpenOption.READ);
467                 DataInputStream din = new DataInputStream(in)) {
468                return din.readShort()==0x1F8B;
469            } catch (IOException ex) {
470                return false;
471            }
472        }
473        
474        /**
475         * Returns the uncompressed size of the file in a quick, but unreliable
476         * manner. It will not report the correct size if:
477         * <ol>
478         * <li>The compressed size is larger than 2<sup>32</sup> bytes.</li>
479         * <li>The file is broken or truncated.</li>
480         * <li>The file has not been generated by a standard-conformant compressor.</li>
481         * <li>It is a multi-volume GZIP stream.</li>
482         * </ol>
483         * <p>
484         * The advantage of this approach is, that it only reads the first 2
485         * and last 4 bytes of the target file. If the first 2 bytes are not
486         * the GZIP magic number, the raw length of the file is returned.
487         * 
488         * @see #isGzipStream(File)
489         * @param file
490         * @return the size of the uncompressed file content.
491         */
492        public static long getGzipStreamSize(File file) {
493            if (!isGzipStream(file)) {
494                return file.length();
495            }
496            RandomAccessFile raf = null;
497            try {
498                raf = new RandomAccessFile(file, "r");
499                if (raf.length() <= 4) {
500                    raf.close();
501                    return file.length();
502                }
503                raf.seek(raf.length() - 4);
504                int b4 = raf.read();
505                int b3 = raf.read();
506                int b2 = raf.read();
507                int b1 = raf.read();
508                return (b1 << 24) + (b2 << 16) + (b3 << 8) + b4;
509            } catch (IOException ex) {
510                return file.length();
511            } finally {
512                if (raf!=null)
513                    try {
514                        raf.close();
515                    } catch (IOException e) {
516                        // ignore
517                    }
518            }
519        }
520    }

522    /**
523     * {@link Session} implementation over {@link ByteBuffer}.
524     */
525    private static final class BufferSession implements Session {
526        private final InputStream in;
527
528        public BufferSession(ByteBuffer buf) {
529            this.in = buf.newInputStream();
530        }
531
532        @Override
533        public void close() throws IOException {
534            in.close();
535        }
536
537        @Override
538        public void skip(long start) throws IOException {
539            while (start > 0) {
540                long diff = in.skip(start);
541                if (diff == 0) {
542                    throw new EOFException("Attempting to read past end of buffer");
543                }
544                start -= diff;
545            }
546        }
547
548        @Override
549        public int read(byte[] buf) throws IOException {
550            return in.read(buf);
551        }
552
553        @Override
554        public int read(byte[] buf, int offset, int length) throws IOException {
555            return in.read(buf,offset,length);
556        }
557    }
