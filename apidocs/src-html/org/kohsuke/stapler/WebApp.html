048/**
049 * Object scoped to the entire webapp. Mostly used for configuring behavior of Stapler.
050 *
051 * <p>
052 * In contrast, {@link Stapler} is a servlet, so there can be multiple instances per webapp.
053 *
054 * @author Kohsuke Kawaguchi
055 * @see WebApp#get(ServletContext)
056 * @see WebApp#getCurrent()  
057 * @see Stapler#getWebApp()
058 */
059public class WebApp {
060    /**
061     * Obtains the {@link WebApp} associated with the given {@link ServletContext}.
062     */
063    public static WebApp get(ServletContext context) {
064        Object o = context.getAttribute(WebApp.class.getName());
065        if(o==null) {
066            synchronized (WebApp.class) {
067                o = context.getAttribute(WebApp.class.getName());
068                if(o==null) {
069                    o = new WebApp(context);
070                    context.setAttribute(WebApp.class.getName(),o);
071                }
072            }
073        }
074        return (WebApp)o;
075    }

164    public WebApp(ServletContext context) {
165        this.context = context;
166        // TODO: allow classloader to be given?
167        facets.addAll(Facet.discoverExtensions(Facet.class, Thread.currentThread().getContextClassLoader(), getClass().getClassLoader()));
168        responseRenderers.add(new HttpResponseRenderer.Default());
169    }

195    public ClassLoader getClassLoader() {
196        ClassLoader cl = classLoader;

204    /**

207    public <T extends Facet> T getFacet(Class<T> type) {
208        for (Facet f : facets)

225    public MetaClass getMetaClass(Klass<?> c) {
226        if(c==null)     return null;
227        synchronized(classMap) {
228            MetaClass mc = classMap.get(c);
229            if(mc==null) {
230                mc = new MetaClass(this,c);
231                classMap.put(c,mc);
232            }
233            return mc;
234        }
235    }

237    /**
238     * Obtains a {@link MetaClass} that represents the type of the given object.
239     *
240     * <p>
241     * This code consults all facets to handle scripting language objects correctly.
242     */
243    public MetaClass getMetaClass(Object o) {
244        return getMetaClass(getKlass(o));
245    }
246
247    public Klass<?> getKlass(Object o) {
248        if (o instanceof KInstance) {
249            KInstance ki = (KInstance) o;
250            Klass k = ki.getKlass();
251            if (k!=null)
252                return k;
253        }
254
255        for (Facet f : facets) {
256            Klass<?> k = f.getKlass(o);
257            if (k!=null)
258                return k;
259        }
260        return Klass.java(o.getClass());
261    }

263    /**
264     * Convenience maintenance method to clear all the cached scripts for the given tearoff type.
265     *
266     * <p>
267     * This is useful when you want to have the scripts reloaded into the live system without
268     * the performance penalty of {@link MetaClass#NO_CACHE}.
269     *
270     * @see MetaClass#NO_CACHE
271     */
272    public void clearScripts(Class<? extends AbstractTearOff> clazz) {
273        synchronized (classMap) {
274            for (MetaClass v : classMap.values()) {
275                AbstractTearOff t = v.getTearOff(clazz);
276                if (t!=null)
277                    t.clearScripts();
278            }
279        }
280    }
281    
282    /**
283     * Convenience maintenance method to clear all the cached information. It will force the MetaClass to be rebuilt.
284     * 
285     * <p>
286     * Take care that the generation of MetaClass information takes a bit of time and so 
287     * this call should not be called too often
288     */
289    public void clearMetaClassCache(){
290        synchronized (classMap) {
291            classMap.clear();
292        }
293    }

297        servlets.put(servletName,servlet);
298    }

300    /**
301     * Gets a reference to some {@link Stapler} servlet in this webapp.
302     *
303     * <p>
304     * Most Stapler webapps will have one {@code <servlet>} entry in web.xml
305     * and if that's the case, that'd be returned. In a fully general case,
306     * a webapp can have multiple servlets and more than one of them can be
307     * {@link Stapler}. This method returns one of those. Which one gets
308     * returned is unspecified.
309     *
310     * <p>
311     * This method is useful if you are in a {@link Filter} and using
312     * Stapler to handle the current request. For example,
313     *
314     * <pre>
315     * WebApp.get(servletContext).getSomeStapler().invoke(
316     *     request,response,
317     *     someJavaObject,
318     *     "/path/to/dispatch/request");
319     * </pre>
320     */
321    public Stapler getSomeStapler() {
322        return servlets.values().iterator().next();
323    }
