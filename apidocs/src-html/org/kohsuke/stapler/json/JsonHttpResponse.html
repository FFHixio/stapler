001package org.kohsuke.stapler.json;
002
003import net.sf.json.JSONObject;
004import org.kohsuke.stapler.HttpResponse;
005import org.kohsuke.stapler.HttpResponses.HttpResponseException;
006import org.kohsuke.stapler.StaplerRequest;
007import org.kohsuke.stapler.StaplerResponse;
008
009import javax.annotation.Nullable;
010import javax.servlet.ServletException;
011import java.io.IOException;
012import java.io.PrintWriter;
013import java.io.StringWriter;
014import java.io.Writer;
015
016/**
017 * {@link JSONObject} as a response.
018 *
019 * @author Carlos Sanchez
020 */
021public class JsonHttpResponse extends HttpResponseException {
022    private final @Nullable JSONObject responseJson;
023    private final int status;
024
025    public JsonHttpResponse(JSONObject o) {
026        this(o, o == null ? 204 : 200);
027    }
028
029    public JsonHttpResponse(JSONObject o, int status) {
030        this.responseJson = o;
031        this.status = status;
032    }
033
034    public JsonHttpResponse(Throwable t, int status) {
035        StringWriter sw = new StringWriter();
036        t.printStackTrace(new PrintWriter(sw));
037        this.responseJson = new JSONObject()
038                .element("error", t.getClass().getName() + ": " + t.getMessage())
039                .element("stackTrace", sw.toString());
040        this.status = status;
041    }
042
043    @Override
044    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException,
045            ServletException {
046        if (status > 0) {
047            rsp.setStatus(status);
048        }
049        if (responseJson != null) {
050            rsp.setContentType("application/json;charset=UTF-8");
051            try (Writer w = rsp.getCompressedWriter(req)) {
052                responseJson.write(w);
053            }
054        }
055    }
056
057}
