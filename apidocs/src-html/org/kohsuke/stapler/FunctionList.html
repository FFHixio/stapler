024package org.kohsuke.stapler;
025
026import java.lang.annotation.Annotation;
027import java.util.*;
028
029/**
030 * Immutable list of {@link Function}s.
031 *
032 * @author Kohsuke Kawaguchi
033 */
034public final class FunctionList extends AbstractList<Function> {
035    private final Function[] functions;
036
037    public FunctionList(Function... functions) {
038        this.functions = functions;
039    }
040
041    public FunctionList(Collection<Function> functions) {
042        this.functions = functions.toArray(new Function[0]);
043    }
044
045    /* internal */ FunctionList filter(Filter f) {
046        List<Function> r = new ArrayList<Function>();
047        for (Function m : functions)
048            if (f.keep(m))
049                r.add(m);
050        return new FunctionList(r.toArray(new Function[0]));
051    }
052
053    @Override
054    public Function get(int index) {
055        return functions[index];
056    }
057
058    public int size() {
059        return functions.length;
060    }
061
062    /**
063     * Compute set unions of two lists.
064     */
065    public FunctionList union(FunctionList that) {
066        Set<Function> combined = new LinkedHashSet<Function>();
067        combined.addAll(Arrays.asList(this.functions));
068        combined.addAll(Arrays.asList(that.functions));
069        return new FunctionList(combined);
070    }
071
072    //public int length() {
073    //    return functions.length;
074    //}
075    //
076    //public Method get(int i) {
077    //    return functions[i];
078    //}
079
080    public interface Filter {
081        boolean keep(Function m);
082
083        Filter ALWAYS_OK = new Filter(){
084            @Override 
085            public boolean keep(Function m) {
086                return true;
087            }
088        };
089    }
090
091    /**
092     * Returns {@link Function}s that start with the given prefix.
093     */
094    public FunctionList prefix(final String prefix) {
095        return filter(new Filter() {
096            public boolean keep(Function m) {
097                return m.getName().startsWith(prefix);
098            }
099        });
100    }
101
102    /**
103     * Returns {@link Function}s that are annotated with the given annotation.
104     */
105    public FunctionList annotated(final Class<? extends Annotation> ann) {
106        return filter(new Filter() {
107            public boolean keep(Function m) {
108                return m.getAnnotation(ann)!=null;
109            }
110        });
111    }
112
113    /**
114     * Returns {@link Function}s that have the given name.
115     */
116    public FunctionList name(final String name) {
117        return filter(new Filter() {
118            public boolean keep(Function m) {
119                return m.getName().equals(name);
120            }
121        });
122    }
123
124    /**
125     * Returns {@link Function}s that has the given type parameters
126     */
127    public FunctionList signature(final Class... args) {
128        return filter(new Filter() {
129            public boolean keep(Function m) {
130                return Arrays.equals(m.getParameterTypes(), args);
131            }
132        });
133    }
134
135    /**
136     * Returns {@link Function}s that are either explicitly {@link WebMethod} or
137     * implicitly so (by having its name start with 'do')
138     */
139    public FunctionList webMethodsLegacy() {
140        return filter(new Filter() {
141            public boolean keep(Function m) {
142                return m.getName().startsWith("do") || m.getAnnotation(WebMethod.class)!=null;
143            }
144        });
145    }
146
147    /**
148     * Returns {@link Function}s that has the parameters
149     * that start with given types (but can have additional parameters.)
150     */
151    public FunctionList signatureStartsWith(final Class... args) {
152        return filter(new Filter() {
153            public boolean keep(Function m) {
154                Class[] params = m.getParameterTypes();
155                if(params.length<args.length)  return false;
156                for( int i=0; i<args.length; i++ ) {
157                    if(params[i]!=args[i])
158                        return false;
159                }
160                return true;
161            }
162        });
163    }
164}
