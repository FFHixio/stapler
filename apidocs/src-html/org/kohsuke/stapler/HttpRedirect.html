024package org.kohsuke.stapler;
025
026import javax.annotation.Nonnull;
027import javax.servlet.ServletException;
028import java.io.IOException;
029
030import static javax.servlet.http.HttpServletResponse.SC_MOVED_TEMPORARILY;
031
032/**
033 * {@link HttpResponse} that dose HTTP 302 redirect.
034 * Extends from {@link RuntimeException} so that you can throw it.
035 *
036 * @author Kohsuke Kawaguchi
037 */
038public final class HttpRedirect extends RuntimeException implements HttpResponse {
039    private final int statusCode;
040    private final String url;
041
042    public HttpRedirect(@Nonnull String url) {
043        this(SC_MOVED_TEMPORARILY,url);
044    }
045
046    public HttpRedirect(int statusCode, @Nonnull String url) {
047        this.statusCode = statusCode;
048        if (url == null) {
049            throw new NullPointerException();
050        }
051        this.url = url;
052    }
053
054    public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
055        rsp.sendRedirect(statusCode,url);
056    }
057
058    /**
059     * @param relative
060     *      The path relative to the context path. The context path + this value
061     *      is sent to the user.
062     * @deprecated
063     *      Use {@link HttpResponses#redirectViaContextPath(String)}.
064     */
065    public static HttpResponse fromContextPath(final String relative) {
066        return HttpResponses.redirectViaContextPath(relative);
067    }
068
069    /**
070     * Redirect to "."
071     */
072    public final static HttpRedirect DOT = new HttpRedirect(".");
073
074    /**
075     * Redirect to the context root
076     */
077    public static HttpResponse CONTEXT_ROOT = fromContextPath("");
078}
