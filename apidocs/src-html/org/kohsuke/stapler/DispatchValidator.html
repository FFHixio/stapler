025package org.kohsuke.stapler;
026
027import javax.annotation.CheckForNull;
028import javax.annotation.Nonnull;
029
030/**
031 * Validates dispatch requests. This validator is configured through {@link WebApp#setDispatchValidator(DispatchValidator)}
032 * and is automatically used by dispatchers created through {@link Facet#createValidatingDispatcher(AbstractTearOff, ScriptExecutor)}.
033 * Extends {@linkplain Facet#createValidatingDispatcher(AbstractTearOff, ScriptExecutor) facet dispatchers} to provide validation
034 * of views before they dispatch, thus allowing a final veto before dispatch begins writing any response body to the
035 * client.
036 *
037 * @see WebApp#setDispatchValidator(DispatchValidator)
038 * @since TODO
039 */
040public interface DispatchValidator {
041
042    /**
043     * Checks if the given request and response should be allowed to dispatch. Returns {@code null} to indicate an
044     * unknown or neutral result.
045     *
046     * @param req the HTTP request to validate
047     * @param rsp the HTTP response
048     * @return true if the request should be dispatched, false if not, or null if unknown or neutral
049     */
050    @CheckForNull Boolean isDispatchAllowed(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp);
051
052    /**
053     * Checks if the given request and response should be allowed to dispatch a view on an optionally present node
054     * object. Returns {@code null} to indicate an unknown or neutral result.
055     *
056     * @param req      the HTTP request to validate
057     * @param rsp      the HTTP response
058     * @param viewName the name of the view to dispatch
059     * @param node     the node being dispatched if present
060     * @return true if the view should be allowed to dispatch, false if it should not, or null if unknown
061     */
062    default @CheckForNull Boolean isDispatchAllowed(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp, @Nonnull String viewName, @CheckForNull Object node) {
063        return isDispatchAllowed(req, rsp);
064    }
065
066    /**
067     * Allows the given request to be dispatched. Further calls to {@link #isDispatchAllowed(StaplerRequest, StaplerResponse)}
068     * should return true for the same request.
069     */
070    void allowDispatch(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp);
071
072    /**
073     * Throws a {@link CancelRequestHandlingException} if the given request is not
074     * {@linkplain #isDispatchAllowed(StaplerRequest, StaplerResponse) allowed}.
075     */
076    default void requireDispatchAllowed(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp) throws CancelRequestHandlingException {
077        Boolean allowed = isDispatchAllowed(req, rsp);
078        if (allowed == null || !allowed) {
079            throw new CancelRequestHandlingException();
080        }
081    }
082
083    /**
084     * Default validator implementation that explicitly allows all dispatch requests to proceed.
085     */
086    DispatchValidator DEFAULT = new DispatchValidator() {
087        @Override
088        public Boolean isDispatchAllowed(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp) {
089            return true;
090        }
091
092        @Override
093        public void allowDispatch(@Nonnull StaplerRequest req, @Nonnull StaplerResponse rsp) {
094            // no-op
095        }
096    };
097}
