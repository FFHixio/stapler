001package org.kohsuke.stapler;
002
003import java.util.AbstractList;
004import java.util.Iterator;
005import java.util.NoSuchElementException;
006
007/**
008 * Single linked list which allows sharing of the suffix.
009 *
010 * @author Kohsuke Kawaguchi
011 * @since 1.220
012 */
013public class SingleLinkedList<T> extends AbstractList<T> {
014    public final T head;
015    public final SingleLinkedList<T> tail;
016
017    public SingleLinkedList(T head, SingleLinkedList<T> tail) {
018        this.head = head;
019        this.tail = tail;
020    }
021
022    /**
023     * Creates a new list by adding a new element as the head.
024     */
025    public SingleLinkedList<T> grow(T item) {
026        return new SingleLinkedList<T>(item,this);
027    }
028
029    @Override
030    public Iterator<T> iterator() {
031        return new Iterator<T>() {
032            SingleLinkedList<T> next = SingleLinkedList.this;
033
034            public boolean hasNext() {
035                return next!=EMPTY_LIST;
036            }
037
038            public T next() {
039                if (!hasNext()) {
040                    throw new NoSuchElementException();
041                }
042                T r = next.head;
043                next = next.tail;
044                return r;
045            }
046
047            public void remove() {
048                throw new UnsupportedOperationException();
049            }
050        };
051    }
052
053    @Override
054    public T get(int index) {
055        return null;
056    }
057
058    @Override
059    public int size() {
060        int sz = 0;
061        for (SingleLinkedList<T> head=this; head!=EMPTY_LIST; head=head.tail)
062            sz++;
063        return sz;
064    }
065
066    @Override
067    public boolean isEmpty() {
068        return this==EMPTY_LIST;
069    }
070
071    public static <T> SingleLinkedList<T> empty() {
072        return EMPTY_LIST;
073    }
074
075    private static final SingleLinkedList EMPTY_LIST = new SingleLinkedList(null,null);
076}
