024package org.kohsuke.stapler;
025
026import com.google.common.cache.Cache;
027import com.google.common.cache.CacheBuilder;
028import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
029
030import java.io.File;
031import java.net.URI;
032import java.net.URISyntaxException;
033import java.net.URL;
034import java.util.logging.Level;
035import java.util.logging.Logger;
036import java.util.regex.Matcher;
037import java.util.regex.Pattern;
038
039/**
040 * Partial default implementation of tear-off class, for convenience of derived classes.
041 *
042 * @param <CLT>
043 *      ClassLoader tear-off.
044 * @author Kohsuke Kawaguchi
045 */
046public abstract class AbstractTearOff<CLT,S,E extends Exception> extends CachingScriptLoader<S,E> {
047
048    private static final Logger LOGGER = Logger.getLogger(AbstractTearOff.class.getName());
049
050    protected final MetaClass owner;
051    protected final CLT classLoader;
052
053    private static final class ExpirableCacheHit<S> {
054        private final long timestamp;
055        private final S script;
056        ExpirableCacheHit(long timestamp, S script) {
057            this.timestamp = timestamp;
058            this.script = script;
059        }
060    }
061    private final Cache<URL, ExpirableCacheHit<S>> cachedScripts = CacheBuilder.newBuilder().softValues().build();
062
063    protected AbstractTearOff(MetaClass owner, Class<CLT> cltClass) {
064        this.owner = owner;
065        if(owner.classLoader!=null)
066            classLoader = owner.classLoader.loadTearOff(cltClass);
067        else
068            classLoader = null;
069    }
070
071    protected final WebApp getWebApp() {
072        return owner.webApp;
073    }
074
075    /**
076     * The file extension of this kind of scripts, such as ".jelly"
077     */
078    protected abstract String getDefaultScriptExtension();
079
080    /**
081     * Checks if the file name is allowed as a script of this type.
082     *
083     * This is necessary to have multiple facets co-exist peacefully
084     * without them trying to load each other's scripts.
085     */
086    protected boolean hasAllowedExtension(String name) {
087        return name.endsWith(getDefaultScriptExtension());
088    }
089
090    /**
091     * Loads the script just from the target class without considering inherited scripts
092     * from its base types.
093     */
094    public S resolveScript(String name) throws E {
095        if (name.lastIndexOf('.')<=name.lastIndexOf('/'))   // no file extension provided
096            name += getDefaultScriptExtension();
097        if (!hasAllowedExtension(name))
098            // for multiple Facets to co-exist peacefully, we need to be able to determine
099            // which Facet is responsible for a given view just from the file name
100            return null;
101
102        URL res = getResource(name);
103        if(res==null) {
104            // look for 'defaults' file
105            int dot = name.lastIndexOf('.');
106            // foo/bar.groovy -> foo/bar.default.groovy
107            // but don't do foo.bar/test -> foo.default.bar/test
108            // as of 2010/9, this behaviour is considered deprecated, but left here for backward compatibility.
109            // we need a better way to refer to the resource of the same name in the base type.
110            if(name.lastIndexOf('/')<dot)
111                res = getResource(name.substring(0, dot) + ".default" + name.substring(dot));
112        }
113        if (res != null) {
114            if (MetaClass.NO_CACHE) {
115                File file = fileOf(res);
116                if (file == null) {
117                    LOGGER.log(Level.FINE, "no file associated with {0}", res);
118                    return parseScript(res);
119                } else {
120                    long timestamp = file.lastModified();
121                    if (timestamp == 0) {
122                        LOGGER.log(Level.FINE, "no timestamp associated with {0}", file);
123                        return parseScript(res);
124                    } else {
125                        ExpirableCacheHit<S> cached = cachedScripts.getIfPresent(res);
126                        if (cached == null) {
127                            S script;
128                            if (LOGGER.isLoggable(Level.FINE)) {
129                                long start = System.nanoTime();
130                                try {
131                                    script = parseScript(res);
132                                } finally {
133                                    LOGGER.log(Level.FINE, "cache miss; took {0}ms to parse {1}", new Object[] {(System.nanoTime() - start) / 1_000_000, res});
134                                }
135                            } else {
136                                LOGGER.log(Level.FINE, "cache miss on {0}", res);
137                                script = parseScript(res);
138                            }
139                            cachedScripts.put(res, new ExpirableCacheHit<>(timestamp, script));
140                            return script;
141                        } else if (timestamp == cached.timestamp) {
142                            LOGGER.log(Level.FINE, "cache hit on {0}", res);
143                            return cached.script;
144                        } else {
145                            LOGGER.log(Level.FINE, "expired cache hit on {0}", res);
146                            S script = parseScript(res);
147                            cachedScripts.put(res, new ExpirableCacheHit<>(timestamp, script));
148                            return script;
149                        }
150                    }
151                }
152            } else {
153                LOGGER.log(Level.FINE, "standard CachingScriptLoader logic applies to {0}", res);
154                return parseScript(res);
155            }
156        }
157
158        return null;
159    }
160
161    private static final Pattern JAR_URL = Pattern.compile("jar:(file:.+)!/.*");
162    @SuppressFBWarnings(value = "PATH_TRAVERSAL_IN", justification = "Files are read from approved plugins, not from user input.")
163    private static File fileOf(URL res) {
164        try {
165            switch (res.getProtocol()) {
166            case "file":
167                return new File(res.toURI());
168            case "jar":
169                Matcher m = JAR_URL.matcher(res.toString());
170                if (m.matches()) {
171                    return new File(new URI(m.group(1)));
172                } else {
173                    return null;
174                }
175            default:
176                return null;
177            }
178        } catch (URISyntaxException | IllegalArgumentException x) {
179            return null; // caching is a best effort
180        }
181    }
182
183    protected final S loadScript(String name) throws E {
184        S s = resolveScript(name);
185        if (s!=null)    return s;
186
187        // not found on this class, delegate to the parent
188        if(owner.baseClass!=null)
189            return ((AbstractTearOff<CLT,S,E>)owner.baseClass.loadTearOff(getClass())).findScript(name);
190
191        return null;
192    }
193
194    /**
195     * Compiles a script into the compiled form.
196     */
197    protected abstract S parseScript(URL res) throws E;
198
199    protected URL getResource(String name) {
200        return owner.klass.getResource(name);
201    }
202}
