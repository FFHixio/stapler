024package org.kohsuke.stapler.tags;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027
028import javax.servlet.RequestDispatcher;
029import javax.servlet.ServletConfig;
030import javax.servlet.ServletContext;
031import javax.servlet.ServletException;
032import javax.servlet.http.HttpServletRequest;
033import javax.servlet.http.HttpServletResponse;
034import javax.servlet.http.HttpServletResponseWrapper;
035import javax.servlet.jsp.JspException;
036import javax.servlet.jsp.PageContext;
037import javax.servlet.jsp.tagext.SimpleTagSupport;
038import java.io.IOException;
039import java.io.PrintWriter;
040
041/**
042 * Includes a side JSP file for the "it" object.
043 *
044 * <p>
045 * This tag looks for a side JSP file of the given name
046 * from the inheritance hierarchy of the it object,
047 * and includes the contents of it, just like {@code <jsp:include>}.
048 *
049 * <p>
050 * For example, if the "it" object is an instance of the <tt>Foo</tt> class,
051 * which looks like the following:
052 *
053 * <pre>{@code
054 * class Foo extends Bar { ... }
055 * class Bar extends Zot { ... }
056 * }</pre>
057 *
058 * <p>
059 * And if you write:
060 * <pre>{@code
061 * <st:include page="abc.jsp"/>
062 * }</pre>
063 * then, it looks for the following files in this order,
064 * and includes the first one found.
065 * <ol>
066 *  <li>a side-file of the Foo class named abc.jsp (/WEB-INF/side-files/Foo/abc.jsp)
067 *  <li>a side-file of the Bar class named abc.jsp (/WEB-INF/side-files/Bar/abc.jsp)
068 *  <li>a side-file of the Zot class named abc.jsp (/WEB-INF/side-files/Zot/abc.jsp)
069 * </ol>
070 *
071 * @author Kohsuke Kawaguchi
072 */
073public class Include extends SimpleTagSupport {
074
075    private Object it;
076
077    private String page;
078
079    /**
080     * Specifies the name of the JSP to be included.
081     */
082    public void setPage(String page) {
083        this.page = page;
084    }
085
086    /**
087     * Specifies the object for which JSP will be included.
088     */
089    public void setIt(Object it) {
090        this.it = it;
091    }
092
093    private Object getPageObject( String name ) {
094        return getJspContext().getAttribute(name,PageContext.PAGE_SCOPE);
095    }
096
097    @SuppressFBWarnings(value = "REQUESTDISPATCHER_FILE_DISCLOSURE", justification = "Forwarding the request to be handled correctly.")
098    public void doTag() throws JspException, IOException {
099        Object it = getJspContext().getAttribute("it",PageContext.REQUEST_SCOPE);
100        final Object oldIt = it;
101        if(this.it!=null)
102            it = this.it;
103
104        ServletConfig cfg = (ServletConfig) getPageObject(PageContext.CONFIG);
105        ServletContext sc = cfg.getServletContext();
106
107        for( Class c = it.getClass(); c!=Object.class; c=c.getSuperclass() ) {
108            String name = "/WEB-INF/side-files/"+c.getName().replace('.','/')+'/'+page;
109            if(sc.getResource(name)!=null) {
110                // Tomcat returns a RequestDispatcher even if the JSP file doesn't exist.
111                // so check if the resource exists first.
112                RequestDispatcher disp = sc.getRequestDispatcher(name);
113                if(disp!=null) {
114                    getJspContext().setAttribute("it",it,PageContext.REQUEST_SCOPE);
115                    try {
116                        HttpServletRequest request = (HttpServletRequest) getPageObject(PageContext.REQUEST);
117                        disp.include(
118                            request,
119                            new Wrapper(
120                                (HttpServletResponse)getPageObject(PageContext.RESPONSE),
121                                new PrintWriter(getJspContext().getOut()) )
122                        );
123                    } catch (ServletException e) {
124                        throw new JspException(e);
125                    } finally {
126                        getJspContext().setAttribute("it",oldIt,PageContext.REQUEST_SCOPE);
127                    }
128                    return;
129                }
130            }
131        }
132
133        throw new JspException("Unable to find '"+page+"' for "+it.getClass());
134    }
135}
136
137class Wrapper extends HttpServletResponseWrapper {
138    private final PrintWriter pw;
139
140    public Wrapper(HttpServletResponse httpServletResponse, PrintWriter w) {
141        super(httpServletResponse);
142        this.pw = w;
143    }
144
145    public PrintWriter getWriter() throws IOException {
146        return pw;
147    }
148}
