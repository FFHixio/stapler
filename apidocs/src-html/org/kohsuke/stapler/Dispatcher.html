024package org.kohsuke.stapler;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027
028import javax.servlet.ServletException;
029import java.io.IOException;
030import java.lang.reflect.InvocationTargetException;
031import java.util.ArrayList;
032import java.util.Arrays;
033import java.util.List;
034import java.util.logging.Level;
035import java.util.logging.Logger;
036
037/**
038 * Controls the dispatching of incoming HTTP requests.
039 *
040 * @author Kohsuke Kawaguchi
041 */
042public abstract class Dispatcher {
043    /**
044     * Tries to handle the given request and returns true
045     * if it succeeds. Otherwise false.
046     *
047     * <p>
048     * We have a few known strategies for handling requests
049     * (for example, one is to try to treat the request as JSP invocation,
050     * another might be try getXXX(), etc) So we use a list of
051     * {@link Dispatcher} and try them one by one until someone
052     * returns true.
053     */
054    public abstract boolean dispatch(RequestImpl req, ResponseImpl rsp, Object node)
055        throws IOException, ServletException, IllegalAccessException, InvocationTargetException;
056
057    /**
058     * Diagnostic string that explains this dispatch rule.
059     */
060    public abstract String toString();
061
062
063    public static boolean traceable() {
064        return TRACE || TRACE_PER_REQUEST || LOGGER.isLoggable(Level.FINE);
065    }
066
067    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node) {
068        trace(req,rsp,String.format("-> evaluate(%s%s,\"%s\")",
069                node==null?"null":'<'+node.toString()+'>',
070                node==null?"":" :"+node.getClass().getName(),
071                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
072    }
073
074    public static void anonymizedTraceEval(StaplerRequest req, StaplerResponse rsp, Object node, String format, String... args) {
075        List<String> arg = new ArrayList<>();
076        arg.add(node == null ? "(null)" : node.getClass().getName());
077        arg.addAll(Arrays.asList(args));
078        EvaluationTrace.ApplicationTracer.trace(req, String.format(format, arg.toArray()));
079    }
080
081    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node, String prefix, String suffix) {
082        trace(req,rsp,String.format("-> evaluate(%s<%s>%s,\"%s\")",
083                prefix,node,suffix,
084                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
085    }
086
087    public static void traceEval(StaplerRequest req, StaplerResponse rsp, Object node, String expression) {
088        trace(req,rsp,String.format("-> evaluate(<%s>.%s,\"%s\")",
089                node,expression,
090                ((RequestImpl)req).tokens.assembleOriginalRestOfPath()));
091    }
092
093    public static void trace(StaplerRequest req, StaplerResponse rsp, String msg, Object... args) {
094        trace(req,rsp,String.format(msg,args));
095    }
096
097    public static void trace(StaplerRequest req, StaplerResponse rsp, String msg) {
098        if(isTraceEnabled(req))
099            EvaluationTrace.get(req).trace(rsp,msg);
100        if(LOGGER.isLoggable(Level.FINE))
101            LOGGER.fine(msg);
102    }
103
104    /**
105     * Checks if tracing is enabled for the given request. Tracing can be
106     * enabled globally with the "stapler.trace=true" system property. Tracing
107     * can be enabled per-request by setting "stapler.trace.per-request=true"
108     * and sending an "X-Stapler-Trace" header set to "true" with the request.
109     */
110    public static boolean isTraceEnabled(StaplerRequest req) {
111        if (TRACE)
112            return true;
113
114        if (TRACE_PER_REQUEST && "true".equals(req.getHeader("X-Stapler-Trace")))
115            return true;
116
117        return false;
118    }
119
120    /**
121     * This flag will activate the evaluation trace.
122     * It adds the evaluation process as HTTP headers,
123     * and when the evaluation failed, special diagnostic 404 page will be rendered.
124     * Useful for developer assistance.
125     */
126    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Used in JTH and others.")
127    public static boolean TRACE = Boolean.getBoolean("stapler.trace");
128
129    /**
130     * This flag will activate the per-request evaluation trace for requests
131     * that have X-Stapler-Trace set to "true".
132     * It adds the evaluation process as HTTP headers,
133     * and when the evaluation failed, special diagnostic 404 page will be rendered.
134     * Useful for developer assistance.
135     */
136    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Legacy switch.")
137    public static boolean TRACE_PER_REQUEST = Boolean.getBoolean("stapler.trace.per-request");
138
139    private static final Logger LOGGER = Logger.getLogger(Dispatcher.class.getName());
140}
