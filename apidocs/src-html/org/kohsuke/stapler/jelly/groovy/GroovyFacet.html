024package org.kohsuke.stapler.jelly.groovy;

026import org.apache.commons.jelly.JellyException;
027import org.kohsuke.MetaInfServices;
028import org.kohsuke.stapler.Dispatcher;
029import org.kohsuke.stapler.Facet;
030import org.kohsuke.stapler.MetaClass;
031import org.kohsuke.stapler.RequestImpl;
032import org.kohsuke.stapler.ResponseImpl;
033import org.kohsuke.stapler.jelly.JellyClassTearOff;
034import org.kohsuke.stapler.jelly.JellyCompatibleFacet;
035import org.kohsuke.stapler.jelly.JellyFacet;
036import org.kohsuke.stapler.jelly.ScriptInvoker;
037import org.kohsuke.stapler.lang.Klass;
038
039import javax.servlet.RequestDispatcher;
040import javax.servlet.ServletException;
041import java.io.IOException;
042import java.util.Collection;
043import java.util.Collections;
044import java.util.List;
045import java.util.Set;
046import java.util.logging.Level;
047
048/**
049 * {@link Facet} that brings in Groovy support on top of Jelly.
050 * 
051 * @author Kohsuke Kawaguchi
052 */
053@MetaInfServices(Facet.class)
054public class GroovyFacet extends Facet implements JellyCompatibleFacet {
055
056    public void buildViewDispatchers(final MetaClass owner, List<Dispatcher> dispatchers) {
057        ScriptInvoker scriptInvoker = owner.webApp.getFacet(JellyFacet.class).scriptInvoker;
058        dispatchers.add(createValidatingDispatcher(owner.loadTearOff(GroovyClassTearOff.class), scriptInvoker));
059        dispatchers.add(createValidatingDispatcher(owner.loadTearOff(GroovyServerPageTearOff.class), scriptInvoker));
060    }
061
062    public Collection<Class<GroovyClassTearOff>> getClassTearOffTypes() {
063        return TEAROFF_TYPES;
064    }
065
066    public Collection<String> getScriptExtensions() {
067        return EXTENSION;
068    }
069
070    public RequestDispatcher createRequestDispatcher(RequestImpl request, Klass type, Object it, String viewName) throws IOException {
071        MetaClass owner = request.getWebApp().getMetaClass(type);
072        ScriptInvoker scriptExecutor = request.getWebApp().getFacet(JellyFacet.class).scriptInvoker;
073        RequestDispatcher d = createRequestDispatcher(owner.loadTearOff(GroovyClassTearOff.class), scriptExecutor, it, viewName);
074        if (d == null) {
075            d = createRequestDispatcher(owner.loadTearOff(GroovyServerPageTearOff.class), scriptExecutor, it, viewName);
076        }
077        return d;
078    }
079
080    @Override
081    public void buildIndexDispatchers(MetaClass owner, List<Dispatcher> dispatchers) {
082        try {
083            if (owner.loadTearOff(JellyClassTearOff.class).findScript("index")!=null) {
084                super.buildIndexDispatchers(owner, dispatchers);
085            }
086        } catch (JellyException e) {
087            LOGGER.log(Level.WARNING, "Failed to parse index.groovy for "+owner, e);
088        }
089    }
090
091    public boolean handleIndexRequest(RequestImpl req, ResponseImpl rsp, Object node, MetaClass nodeMetaClass) throws IOException, ServletException {
092        ScriptInvoker scriptExecutor = req.getWebApp().getFacet(JellyFacet.class).scriptInvoker;
093        return handleIndexRequest(nodeMetaClass.loadTearOff(GroovyClassTearOff.class), scriptExecutor, req, rsp, node) ||
094                handleIndexRequest(nodeMetaClass.loadTearOff(GroovyServerPageTearOff.class), scriptExecutor, req, rsp, node);
095    }
096
097    private static final Set<Class<GroovyClassTearOff>> TEAROFF_TYPES = Collections.singleton(GroovyClassTearOff.class);
098
099    private static final Set<String> EXTENSION = Collections.singleton(".groovy");
100}
