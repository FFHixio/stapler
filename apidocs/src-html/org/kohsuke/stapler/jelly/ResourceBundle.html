024package org.kohsuke.stapler.jelly;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027import org.kohsuke.stapler.MetaClass;
028import org.kohsuke.stapler.WebApp;
029
030import java.io.IOException;
031import java.io.InputStream;
032import java.io.StringWriter;
033import java.net.URL;
034import java.text.MessageFormat;
035import java.util.Collections;
036import java.util.HashMap;
037import java.util.Locale;
038import java.util.Map;
039import java.util.Properties;
040import java.util.concurrent.ConcurrentHashMap;
041
042/**
043 * Cache of localization strings.
044 * 
045 * @author Kohsuke Kawaguchi
046 */
047public class ResourceBundle {
048    /**
049     * URL to load message resources from, except the ".properties" suffix.
050     * <p>
051     * This is normally something like <tt>file://path/to/somewhere/org/acme/Foo</tt>.
052     */
053    private final String baseName;
054
055    /**
056     * Loaded messages.
057     */
058    private final Map<String,Properties> resources = new ConcurrentHashMap<String,Properties>();
059
060    public ResourceBundle(String baseName) {
061        this.baseName = baseName;
062    }
063
064    public String getBaseName() {
065        return baseName;
066    }
067
068    public String format(Locale locale, String key, Object... args) {
069        String str = getFormatString(locale, key);
070        if(str==null)
071            // see http://www.nabble.com/i18n-and-l10n-problems-td16004047.html for more discussion
072            // return MessageFormat.format(key,args);
073            return key;
074
075        return MessageFormat.format(str,args);
076    }
077
078    /**
079     * Gets the format string for the given key.
080     * <p>
081     * This method performs a search so that a look up for "pt_BR" would delegate
082     * to "pt" then "" (the no-locale locale.)
083     */
084    public String getFormatString(Locale locale, String key) {
085        String[] suffixes = toStrings(locale);
086
087        while(true) {
088            for (int i=0; i<suffixes.length; i++) {
089                String suffix = suffixes[i];
090                String msg = get(suffix).getProperty(key);
091                if(msg!=null && msg.length()>0)
092                    // ignore a definition without value, because stapler:i18n generates
093                    // value-less definitions
094                    return msg;
095
096                int idx = suffix.lastIndexOf('_');
097                if(idx<0)   // failed to find
098                    return null;
099                suffixes[i] = suffix.substring(0,idx);
100            }
101        }
102    }
103
104    /**
105     * Works like {@link #getFormatString(Locale, String)} except there's no
106     * searching up the delegation chain.
107     */
108    public String getFormatStringWithoutDefaulting(Locale locale, String key) {
109        for (String s : toStrings(locale)) {
110            String msg = get(s).getProperty(key);
111            if(msg!=null && msg.length()>0)
112                return msg;
113        }
114        return null;
115    }
116
117    /**
118     * Some language codes have changed over time, such as Hebrew from iw to he.
119     * This method returns all such variations in an array.
120     *
121     * @see Locale#getLanguage()
122     */
123    private String[] toStrings(Locale l) {
124        String v = ISO639_MAP.get(l.getLanguage());
125        if (v==null)
126            return new String[]{'_'+l.toString()};
127        else
128            return new String[]{'_'+l.toString(),
129                                '_'+v+l.toString().substring(2)};
130    }
131
132    protected void clearCache() {
133        resources.clear();
134    }
135
136    protected Properties get(String key) {
137        Properties props;
138
139        if(!MetaClass.NO_CACHE) {
140            props = resources.get(key);
141            if(props!=null)     return props;
142        }
143
144        // attempt to load
145        props = new Properties();
146        String url = baseName + key + ".properties";
147        InputStream in=null;
148        try {
149            in = openStream(url);
150            // an user reported that on IBM JDK, URL.openStream
151            // returns null instead of IOException.
152            // see http://www.nabble.com/WAS---Hudson-tt16026561.html
153        } catch (IOException e) {
154            // failed.
155        }
156
157        if(in!=null) {
158            try {
159                try {
160                    props.load(in);
161                } finally {
162                    in.close();
163                }
164            } catch (IOException e) {
165                throw new Error("Failed to load "+url,e);
166            }
167        }
168
169        resources.put(key,wrapUp(key.length()>0 ? key.substring(1) : "",props));
170        return props;
171    }
172
173    @SuppressFBWarnings(value = "URLCONNECTION_SSRF_FD", justification = "Not relevant in this situation.")
174    private static InputStream openStream(String url) throws IOException {
175        return new URL(url).openStream();
176    }
177
178    /**
179     * Interception point for property loading.
180     */
181    protected Properties wrapUp(String locale, Properties props) {
182        return props;
183    }
184
185    @Override
186    public boolean equals(Object o) {
187        if (this == o) return true;
188        if (o == null || getClass() != o.getClass()) return false;
189
190        ResourceBundle that = (ResourceBundle) o;
191        return baseName.equals(that.baseName);
192    }
193
194    @Override
195    public int hashCode() {
196        return baseName.hashCode();
197    }
198
199    public static ResourceBundle load(URL jellyUrl) {
200        return load(jellyUrl.toExternalForm());
201    }
202
203    /**
204     * Loads the resource bundle associated with the Jelly script.
205     */
206    public static ResourceBundle load(String jellyUrl) {
207        if(jellyUrl.endsWith(".jelly"))    // cut the trailing .jelly
208            jellyUrl = jellyUrl.substring(0,jellyUrl.length()-".jelly".length());
209
210        JellyFacet facet = WebApp.getCurrent().getFacet(JellyFacet.class);
211        return facet.resourceBundleFactory.create(jellyUrl);
212    }
213
214    /**
215     * JDK internally converts new ISO-639 code back to old code. This table provides reverse mapping.
216     */
217    private static final Map<String,String> ISO639_MAP = new HashMap<String, String>();
218
219    static {
220        ISO639_MAP.put("iw","he");
221        ISO639_MAP.put("ji","yi");
222        ISO639_MAP.put("in","id");
223    }
224}
