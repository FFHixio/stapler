024package org.kohsuke.stapler.jelly;

026import org.apache.commons.jelly.JellyException;
027import org.apache.commons.jelly.Script;
028import org.kohsuke.stapler.AbstractTearOff;
029import org.kohsuke.stapler.Dispatcher;
030import org.kohsuke.stapler.Facet;
031import org.kohsuke.stapler.MetaClass;
032import org.kohsuke.stapler.StaplerRequest;
033import org.kohsuke.stapler.StaplerResponse;
034
035import javax.servlet.RequestDispatcher;
036import javax.servlet.ServletException;
037import java.io.IOException;
038import java.net.URL;
039import java.util.logging.Logger;
040
041import static org.kohsuke.stapler.Dispatcher.trace;
042import static org.kohsuke.stapler.Dispatcher.traceable;
043
044/**
045 * @author Kohsuke Kawaguchi
046 */
047public class JellyClassTearOff extends AbstractTearOff<JellyClassLoaderTearOff,Script,JellyException> {
048    private JellyFacet facet;
049
050    public JellyClassTearOff(MetaClass owner) {
051        super(owner,JellyClassLoaderTearOff.class);
052        facet = owner.webApp.getFacet(JellyFacet.class);
053    }
054
055    protected Script parseScript(URL res) throws JellyException {
056        return new JellyViewScript(owner.klass, res, classLoader.createContext().compileScript(res));
057    }
058
059    @Override
060    protected String getDefaultScriptExtension() {
061        return ".jelly";
062    }
063
064    @Override
065    protected boolean hasAllowedExtension(String name) {
066        return name.endsWith(".jelly") || name.endsWith(".jellytag");
067    }
068
069    /**
070     * Aside from looking into our own, consult other facets that can handle Jelly-compatible scripts.
071     */
072    @Override
073    public Script resolveScript(String name) throws JellyException {
074        // cut off the extension so that we can search all the extensions
075        String shortName;
076        int dot = name.lastIndexOf('.');
077        if (dot>name.lastIndexOf('/'))      shortName = name.substring(0, dot);
078        else                                shortName = name;
079
080        for (Facet f : owner.webApp.facets) {
081            if (f instanceof JellyCompatibleFacet && !(f instanceof JellyFacet)) {
082                JellyCompatibleFacet jcf = (JellyCompatibleFacet) f;
083                for (Class<? extends AbstractTearOff<?,? extends Script,?>> ct : jcf.getClassTearOffTypes()) {
084                    try {
085                        Script s = owner.loadTearOff(ct).resolveScript(shortName);
086                        if (s!=null)    return s;
087                    } catch (Exception e) {
088                        throw new JellyException("Failed to load "+shortName+" from "+jcf,e);
089                    }
090                }
091            }
092        }
093
094        return super.resolveScript(name);
095    }
096
097    /**
098     * Serves <tt>index.jelly</tt> if it's available, and returns true.
099     */
100    @Deprecated
101    public boolean serveIndexJelly(StaplerRequest req, StaplerResponse rsp, Object node) throws ServletException, IOException {
102        try {
103            Script script = findScript("index.jelly");
104            if(script!=null) {
105                String src = "index.jelly";
106                if (script instanceof JellyViewScript) {
107                    JellyViewScript jvs = (JellyViewScript) script;
108                    src = jvs.getName();
109                }
110                Dispatcher.anonymizedTraceEval(req, rsp, node, "%s: Jelly index: %s", src);
111                if(traceable()) {
112                    trace(req,rsp,"-> %s on <%s>",src,node);
113                }
114                facet.scriptInvoker.invokeScript(req, rsp, script, node);
115                return true;
116            }
117            return false;
118        } catch (JellyException e) {
119            throw new ServletException(e);
120        }
121    }
122
123    /**
124     * Creates a {@link RequestDispatcher} that forwards to the jelly view, if available.
125     */
126    @Deprecated
127    public RequestDispatcher createDispatcher(Object it, String viewName) throws IOException {
128        try {
129            // backward compatible behavior that expects full file name including ".jelly"
130            Script script = findScript(viewName);
131            if(script!=null)
132                return new JellyRequestDispatcher(it,script);
133            
134            // this is what the look up was really supposed to be.
135            script = findScript(viewName+".jelly");
136            if(script!=null)
137                return new JellyRequestDispatcher(it,script);
138            return null;
139        } catch (JellyException e) {
140            IOException io = new IOException(e.getMessage());
141            io.initCause(e);
142            throw io;
143        }
144    }
145
146    private static final Logger LOGGER = Logger.getLogger(JellyClassTearOff.class.getName());
147}
