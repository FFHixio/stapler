024package org.kohsuke.stapler.jelly;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027import org.apache.commons.jelly.JellyContext;
028import org.apache.commons.jelly.JellyException;
029import org.apache.commons.jelly.JellyTagException;
030import org.apache.commons.jelly.Tag;
031import org.apache.commons.jelly.TagLibrary;
032import org.apache.commons.jelly.XMLOutput;
033import org.apache.commons.jelly.expression.ConstantExpression;
034import org.apache.commons.jelly.expression.Expression;
035import org.apache.commons.jelly.impl.ExpressionAttribute;
036import org.apache.commons.jelly.impl.StaticTag;
037import org.apache.commons.jelly.impl.StaticTagScript;
038import org.apache.commons.jelly.impl.TagScript;
039import org.xml.sax.Attributes;
040import org.xml.sax.SAXException;
041import org.xml.sax.helpers.AttributesImpl;
042
043/**
044 * Jelly tag library for static tags.
045 *
046 * <p>
047 * Unlike {@link StaticTagScript}, this doesn't even try to see if the tag name is available as a dynamic tag.
048 * By not doing so, this implementation achieves a better performance both in speed and memory usage. 
049 *
050 * <p>
051 * Jelly by default uses {@link StaticTagScript} instance to represent a tag that's parsed as a static tag,
052 * and for each invocation, this code checks if the tag it represents is now defined as a dynamic tag.
053 * Plus it got the code to cache {@link StaticTag} instances per thread, which consumes more space and time.
054 *
055 * @author Kohsuke Kawaguchi
056 * @since 1.342
057 */
058public class ReallyStaticTagLibrary extends TagLibrary {
059    /**
060     * IIUC, this method will never be invoked.
061     */
062    @Override
063    public Tag createTag(final String name, Attributes attributes) throws JellyException {
064        return null;
065    }
066
067    @Override
068    public TagScript createTagScript(String tagName, Attributes atts) throws JellyException {
069        return createTagScript();
070    }
071
072    /**
073     * Creates a new instance of {@link TagScript} that generates a literal element.
074     */
075    public static TagScript createTagScript() {
076        return new TagScript() {
077            /**
078             * If all the attributes are constant, as is often the case with literal tags,
079             * then we can skip the attribute expression evaluation altogether.
080             */
081            private boolean allAttributesAreConstant = true;
082
083            @Override
084            public void addAttribute(String name, Expression expression) {
085                allAttributesAreConstant &= expression instanceof ConstantExpression;
086                super.addAttribute(name, expression);
087            }
088
089            @Override
090            public void addAttribute(String name, String prefix, String nsURI, Expression expression) {
091                allAttributesAreConstant &= expression instanceof ConstantExpression;
092                super.addAttribute(name, prefix, nsURI, expression);
093            }
094
095            public void run(JellyContext context, XMLOutput output) throws JellyTagException {
096                Attributes actual = (allAttributesAreConstant && !EMIT_LOCATION) ? getSaxAttributes() : buildAttributes(context);
097
098                try {
099                    output.startElement(getNsUri(),getLocalName(),getElementName(),actual);
100                    getTagBody().run(context,output);
101                    output.endElement(getNsUri(),getLocalName(),getElementName());
102                } catch (SAXException x) {
103                    throw new JellyTagException(x);
104                }
105            }
106
107            private AttributesImpl buildAttributes(JellyContext context) {
108                AttributesImpl actual = new AttributesImpl();
109
110                for (ExpressionAttribute att : attributes.values()) {
111                    Expression expression = att.exp;
112                    String v = expression.evaluateAsString(context);
113                    if (v==null)    continue; // treat null as no attribute
114                    actual.addAttribute(att.nsURI, att.name, att.qname(),"CDATA", v);
115                }
116
117                if (EMIT_LOCATION) {
118                    actual.addAttribute("","file","file","CDATA",String.valueOf(getFileName()));
119                    actual.addAttribute("","line","line","CDATA",String.valueOf(getLineNumber()));
120
121                    // try to obtain the meaningful part of the script and put it in CSS with a
122                    // class name like "jelly-foo-bar-xyz" given "file://path/to/src/tree/src/main/resources/foo/bar/xyz.jelly"
123                    String form = getFileName().replace('\\','/');
124                    for (String suffix : SUFFIX) {
125                        int idx = form.lastIndexOf(suffix);
126                        if (idx>0)  form=form.substring(idx+suffix.length());
127                    }
128
129                    int c = actual.getIndex("class");
130                    if (c>=0)   actual.setValue(c, actual.getValue(c)+" "+form);
131                    else        actual.addAttribute("","class","class","CDATA",form);
132                }
133
134                return actual;
135            }
136        };
137    }
138    /**
139     * Reusable instance.
140     */
141    public static final TagLibrary INSTANCE = new ReallyStaticTagLibrary();
142
143    /**
144     * If true, emit the location information.
145     */
146    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Legacy switch.")
147    public static boolean EMIT_LOCATION = false;
148
149    private static final String[] SUFFIX = {"src/main/resources/","src/test/resources/"};
150}
