024package org.kohsuke.stapler.jelly;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027import org.apache.commons.jelly.JellyTagException;
028import org.apache.commons.jelly.XMLOutput;
029import org.xml.sax.SAXException;
030import org.jvnet.maven.jellydoc.annotation.NoContent;
031
032import java.io.File;
033import java.io.FileInputStream;
034import java.io.FileNotFoundException;
035import java.io.FileReader;
036import java.io.IOException;
037import java.io.InputStream;
038import java.io.InputStreamReader;
039import java.io.Reader;
040import java.net.URL;
041import java.nio.charset.StandardCharsets;
042
043/**
044 * Copies a stream as text.
045 * @author Kohsuke Kawaguchi
046 */
047@NoContent
048public class CopyStreamTag extends AbstractStaplerTag {
049    private Reader in;
050
051    public void setReader(Reader in) {
052        this.in = in;
053    }
054
055    public void setInputStream(InputStream in) {
056        this.in = new InputStreamReader(in, StandardCharsets.UTF_8);
057    }
058
059    public void setFile(File f) throws FileNotFoundException {
060        this.in = new InputStreamReader(new FileInputStream(f), StandardCharsets.UTF_8);
061    }
062
063    @SuppressFBWarnings(value = "URLCONNECTION_SSRF_FD", justification = "Not relevant in this situation.")
064    public void setUrl(URL url) throws IOException {
065        setInputStream(url.openStream());
066    }
067
068
069    public void doTag(XMLOutput xmlOutput) throws JellyTagException {
070        if(in==null)
071            // In JEXL, failures evaluate to null, so if the input is meant to be
072            // set from expression, we don't want that evaluation failure to cause
073            // the entire page rendering to fail.
074            return;
075
076        char[] buf = new char[8192];
077        int len;
078
079        try {
080            try {
081                while((len=in.read(buf,0,buf.length))>=0) {
082                    int last = 0;
083                    for (int i=0; i<len; i++ ) {
084                        char ch = buf[i];
085                        if (ch == '<') {
086                            xmlOutput.characters(buf, last, i - last); // flush
087                            xmlOutput.characters(CHARS_LE, 0, CHARS_LE.length);
088                            last = i + 1;
089                        } else if (ch == '&') {
090                            xmlOutput.characters(buf,last,i-last); // flush
091                            xmlOutput.characters(CHARS_AMP,0,CHARS_AMP.length);
092                            last = i+1;
093                        }
094                    }
095                    xmlOutput.characters(buf,last,len-last);
096                }
097            } finally {
098                in.close();
099            }
100        } catch (IOException e) {
101            throw new JellyTagException(e);
102        } catch (SAXException e) {
103            throw new JellyTagException(e);
104        }
105    }
106
107    private static final char[] CHARS_LE = "&lt;".toCharArray();
108    private static final char[] CHARS_AMP = "&amp;".toCharArray();
109}
