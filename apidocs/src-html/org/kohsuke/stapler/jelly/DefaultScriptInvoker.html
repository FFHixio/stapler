045/**
046 * Standard implementation of {@link ScriptInvoker}.
047 * 
048 * @author Kohsuke Kawaguchi
049 */
050public class DefaultScriptInvoker implements ScriptInvoker, XMLOutputFactory {
051    public void invokeScript(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException, JellyTagException {
052        XMLOutput xmlOutput = createXMLOutput(req, rsp, script, it);
053
054        invokeScript(req,rsp,script,it,xmlOutput);
055        
056        xmlOutput.flush();
057        xmlOutput.close();
058    }
059
060    public void invokeScript(StaplerRequest req, StaplerResponse rsp, Script script, Object it, XMLOutput out) throws IOException, JellyTagException {
061        JellyContext context = createContext(req,rsp,script,it);
062        exportVariables(req, rsp, script, it, context);
063
064        script.run(context,out);
065    }
066
067    protected XMLOutput createXMLOutput(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException {
068        // TODO: make XMLOutput auto-close OutputStream to avoid leak
069        String ct = rsp.getContentType();
070        XMLOutput output;
071        if (ct != null && !ct.startsWith("text/html")) {
072            output = XMLOutput.createXMLOutput(createOutputStream(req, rsp, script, it));
073        } else {
074            output = HTMLWriterOutput.create(createOutputStream(req, rsp, script, it));
075
076        }
077        return output;
078    }
079
080    private boolean doCompression(Script script) {
081        if (COMPRESS_BY_DEFAULT)    return true;
082        if (script instanceof TagScript) {
083            TagScript ts = (TagScript) script;
084            if(ts.getLocalName().equals("compress"))
085                return true;
086        }
087        return false;
088    }
089
090    private interface OutputStreamSupplier {
091        @Nonnull OutputStream get() throws IOException;
092    }
093
094    private static class LazyOutputStreamSupplier implements OutputStreamSupplier {
095        private final OutputStreamSupplier supplier;
096        private volatile OutputStream out;
097
098        private LazyOutputStreamSupplier(OutputStreamSupplier supplier) {
099            this.supplier = supplier;
100        }
101
102        @Override
103        @Nonnull
104        public OutputStream get() throws IOException {
105            if (out == null) {
106                synchronized (this) {
107                    if (out == null) {
108                        out = supplier.get();
109                    }
110                }
111            }
112            return out;
113        }
114    }
115
116    protected OutputStream createOutputStream(StaplerRequest req, StaplerResponse rsp, Script script, Object it) throws IOException {
117        OutputStreamSupplier out = new LazyOutputStreamSupplier(() -> {
118            req.getWebApp().getDispatchValidator().requireDispatchAllowed(req, rsp);
119            return doCompression(script) ? rsp.getCompressedOutputStream(req) : new BufferedOutputStream(rsp.getOutputStream());
120        });
121        return new OutputStream() {
122            @Override
123            public void write(int b) throws IOException {
124                out.get().write(b);
125            }
126
127            @Override
128            public void write(byte[] b) throws IOException {
129                out.get().write(b);
130            }
131
132            @Override
133            public void write(byte[] b, int off, int len) throws IOException {
134                out.get().write(b, off, len);
135            }
136
137            @Override
138            public void flush() throws IOException {
139                // flushing ServletOutputStream causes Tomcat to
140                // send out headers, making it impossible to set contentType from the script.
141                // so don't let Jelly flush.
142            }
143
144            @Override
145            public void close() throws IOException {
146                out.get().close();
147            }
148        };
149    }
150
151    protected void exportVariables(StaplerRequest req, StaplerResponse rsp, Script script, Object it, JellyContext context) {
152        Enumeration en = req.getAttributeNames();
153        // expose request attributes, just like JSP
154        while (en.hasMoreElements()) {
155            String name = (String) en.nextElement();
156            context.setVariable(name,req.getAttribute(name));
157        }
158
159        context.setVariable("request",req);
160        context.setVariable("response",rsp);
161        context.setVariable("it",it);
162        ServletContext servletContext = req.getServletContext();
163        context.setVariable("servletContext",servletContext);
164        context.setVariable("app",servletContext.getAttribute("app"));
165        // property bag to store request scope variables
166        context.setVariable("requestScope",context.getVariables());
167        // this variable is needed to make "jelly:fmt" taglib work correctly
168        context.setVariable("org.apache.commons.jelly.tags.fmt.locale",req.getLocale());
169    }
170
171    protected JellyContext createContext(final StaplerRequest req, StaplerResponse rsp, Script script, Object it) {
172        CustomJellyContext context = new CustomJellyContext();
173        // let Jelly see the whole classes
174        context.setClassLoader(req.getWebApp().getClassLoader());
175        // so TagScript.getBodyText() will use HTMLWriterOutput
176        context.setVariable(XMLOutputFactory.class.getName(), this);
177        return context;
178    }
179
180    public XMLOutput createXMLOutput(Writer writer, boolean escapeText) {
181        StaplerResponse rsp = Stapler.getCurrentResponse();
182        String ct = rsp!=null ? rsp.getContentType() : "?";
183        if (ct != null && !ct.startsWith("text/html"))
184            return XMLOutput.createXMLOutput(writer, escapeText);
185        return HTMLWriterOutput.create(writer, escapeText);
186    }
187
188    /**
189     * Whether gzip compression of the dynamic content is enabled by default or not.
190     *
191     * <p>
192     * For non-trivial web applications, where the performance matters, it is normally a good trade-off to spend
193     * a bit of CPU cycles to compress data. This is because:
194     *
195     * <ul>
196     * <li>CPU is already 1 or 2 order of magnitude faster than RAM and network.
197     * <li>CPU is getting faster than any other components, such as RAM and network.
198     * <li>Because of the TCP window slow start, on a large latency network, compression makes difference in
199     *     the order of 100ms to 1sec to the completion of a request by saving multiple roundtrips.
200     * </ul>
201     *
202     * Stuff rendered by Jelly is predominantly text, so the compression would work well.
203     *
204     * @see <a href="http://www.slideshare.net/guest22d4179/latency-trumps-all">Latency Trumps All</a>
205     */
206    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Legacy switch.")
207    public static boolean COMPRESS_BY_DEFAULT = Boolean.parseBoolean(System.getProperty(DefaultScriptInvoker.class.getName()+".compress","true"));
208}
