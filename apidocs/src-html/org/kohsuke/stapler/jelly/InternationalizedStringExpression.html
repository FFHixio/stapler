024package org.kohsuke.stapler.jelly;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027import org.apache.commons.jelly.expression.Expression;
028import org.apache.commons.jelly.expression.ExpressionSupport;
029import org.apache.commons.jelly.JellyException;
030import org.apache.commons.jelly.JellyContext;
031import org.jvnet.localizer.LocaleProvider;
032import org.kohsuke.stapler.Stapler;
033
034import java.util.List;
035import java.util.ArrayList;
036import java.util.Collections;
037import java.util.Arrays;
038
039/**
040 * Expression of the form "%messageName(arg1,arg2,...)" that represents
041 * internationalized text.
042 *
043 * <p>
044 * The "(arg1,...)" portion is optional and can be omitted. Each argument
045 * is assumed to be a parenthesis-balanced expression and passed to
046 * {@link JellyClassLoaderTearOff#EXPRESSION_FACTORY} to be parsed.
047 *
048 * <p>
049 * The message resource is loaded from files like "xyz.properties" and
050 * "xyz_ja.properties" when the expression is placed in "xyz.jelly". 
051 *
052 *
053 * @author Kohsuke Kawaguchi
054 */
055public class InternationalizedStringExpression extends ExpressionSupport {
056    public final ResourceBundle resourceBundle;
057    private final Expression[] arguments;
058    public final String key;
059    public final String expressionText;
060
061    public InternationalizedStringExpression(ResourceBundle resourceBundle, String text) throws JellyException {
062        this.resourceBundle = resourceBundle;
063        this.expressionText = text;
064        if(!text.startsWith("%"))
065            throw new JellyException(text+" doesn't start with %");
066        text = text.substring(1);
067
068        int idx = text.indexOf('(');
069        if(idx<0) {
070            // no arguments
071            key = text;
072            arguments = EMPTY_ARGUMENTS;
073            return;
074        }
075
076        List<Expression> args = new ArrayList<Expression>();
077        key = text.substring(0,idx);
078        text = text.substring(idx+1);   // at this point text="arg,arg)"
079        while(text.length()>0) {
080            String token = tokenize(text);
081            args.add(JellyClassLoaderTearOff.EXPRESSION_FACTORY.createExpression(token));
082            text = text.substring(token.length()+1);
083        }
084
085        this.arguments = args.toArray(new Expression[args.size()]);
086    }
087    
088    public List<Expression> getArguments() {
089        return Collections.unmodifiableList(Arrays.asList(arguments));
090    }
091
092    /**
093     * Takes a string like "arg)" or "arg,arg,...)", then
094     * find "arg" and returns it.
095     *
096     * Note: this code is also copied into idea-stapler-plugin,
097     * so don't forget to update that when this code changes.
098     */
099    @SuppressFBWarnings(value = "SF_SWITCH_NO_DEFAULT", justification = "No default needed.")
100    private String tokenize(String text) throws JellyException {
101        int parenthesis=0;
102        for(int idx=0;idx<text.length();idx++) {
103            char ch = text.charAt(idx);
104            switch (ch) {
105            case ',':
106                if(parenthesis==0)
107                    return text.substring(0,idx);
108                break;
109            case '(':
110            case '{':
111            case '[':
112                parenthesis++;
113                break;
114            case ')':
115                if(parenthesis==0)
116                    return text.substring(0,idx);
117                // fall through
118            case '}':
119            case ']':
120                parenthesis--;
121                break;
122            case '"':
123            case '\'':
124                // skip strings
125                idx = text.indexOf(ch,idx+1);
126                break;
127            }
128        }
129        throw new JellyException(expressionText+" is missing ')' at the end");
130    }
131
132    public String getExpressionText() {
133        return expressionText;
134    }
135
136    public Object evaluate(JellyContext context) {
137        return format(evaluateArguments(context));
138    }
139
140    private Object format(Object[] args) {
141        // notify the listener if set
142        InternationalizedStringExpressionListener listener = (InternationalizedStringExpressionListener) Stapler.getCurrentRequest().getAttribute(LISTENER_NAME);
143        if(listener!=null)
144            listener.onUsed(this, args);
145
146        return resourceBundle.format(LocaleProvider.getLocale(), key, args);
147    }
148
149    private Object[] evaluateArguments(JellyContext jellyContext) {
150        Object[] args = new Object[arguments.length];
151        for (int i = 0; i < args.length; i++)
152            args[i] = arguments[i].evaluate(jellyContext);
153        return args;
154    }
155
156    /**
157     * Creates a new {@link Expression} that performs proper HTML escaping.
158     */
159    public Expression makeEscapingExpression() {
160        return new ExpressionSupport() {
161            public String getExpressionText() {
162                return expressionText;
163            }
164
165            public Object evaluate(JellyContext context) {
166                Object[] args = evaluateArguments(context);
167                args = Stapler.htmlSafeArguments(args);
168                return format(args);
169            }
170        };
171    }
172
173    /**
174     * @deprecated Use {@link org.kohsuke.stapler.RawHtmlArgument}
175     */
176    public static final class RawHtmlArgument extends org.kohsuke.stapler.RawHtmlArgument {
177        public RawHtmlArgument(Object value) {
178            super(value);
179        }
180    }
181
182    private static final Expression[] EMPTY_ARGUMENTS = new Expression[0];
183    private static final String LISTENER_NAME = InternationalizedStringExpressionListener.class.getName();
184}
