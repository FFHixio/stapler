024package org.kohsuke.stapler;
025
026import javax.annotation.CheckForNull;
027import javax.annotation.Nonnull;
028import java.net.URL;
029import java.util.ArrayList;
030import java.util.List;
031import java.util.ServiceLoader;
032import java.util.logging.Level;
033import java.util.logging.Logger;
034
035/**
036 * Service provider interface allowing to hook into webapp resource lookup.
037 *
038 * This cannot be made a property of WebApp as other behavior customizations, as webapp resource lookup is done before we have StaplerRequest/StaplerResponse.
039 */
040public abstract class LocaleDrivenResourceProvider {
041    /**
042     * Returns the URL corresponding to the specified resource path.
043     *
044     * The path can take two forms:
045     *
046     * <ul>
047     *     <li>A full URL represented as string. In Jenkins this is typically plugin webapp resources.</li>
048     *     <li>A path with leading slash. These are actual webapp resources, with that being the implicit base directory.</li>
049     * </ul>
050     *
051     * @param path the path to the resource
052     * @return the URL to the file, if it is to be overridden, null otherwise.
053     */
054    @CheckForNull
055    abstract public URL lookup(@Nonnull String path);
056
057    private static List<LocaleDrivenResourceProvider> localeDrivenResourceProviders;
058
059    private static final Logger LOGGER = Logger.getLogger(LocaleDrivenResourceProvider.class.getName());
060
061    private static synchronized List<LocaleDrivenResourceProvider> getLocaleDrivenResourceProviders() {
062        if (localeDrivenResourceProviders == null) {
063            localeDrivenResourceProviders = new ArrayList<>();
064            for (LocaleDrivenResourceProvider provider : ServiceLoader.load(LocaleDrivenResourceProvider.class)) {
065                localeDrivenResourceProviders.add(provider);
066                LOGGER.fine(() -> "Registered LocaleDrivenResourceProvider: " + provider);
067            }
068        }
069        return localeDrivenResourceProviders;
070    }
071
072    /* package */ static URL lookupResource(String path) {
073        for (LocaleDrivenResourceProvider provider : getLocaleDrivenResourceProviders()) {
074            try {
075                URL url = provider.lookup(path);
076                if (url != null) {
077                    return url;
078                }
079            } catch (Exception e) {
080                LOGGER.log(Level.WARNING, "Failed to look up URL for " + path + " from " + provider, e);
081            }
082        }
083        return null;
084    }
085}
