024package org.kohsuke.stapler;
025
026import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
027
028import java.lang.reflect.InvocationTargetException;
029import java.util.HashMap;
030import java.util.Map;
031
032/**
033 * Allows "tear-off" objects to be linked to the parent object.
034 *
035 * <p>
036 * This mechanism is used to avoid static linking optional packages,
037 * so that stapler can work even when the optional dependencies are missing.
038 *
039 * @author Kohsuke Kawaguchi
040 */
041@SuppressFBWarnings(value = "UG_SYNC_SET_UNSYNC_GET", justification = "Get is intentionally unsynchronized.")
042public abstract class TearOffSupport {
043    private volatile Map<Class,Object> tearOffs;
044
045    public final <T> T getTearOff(Class<T> t) {
046        Map<Class,Object> m = tearOffs;
047        if(m==null)     return null;
048        return (T)m.get(t);
049    }
050
051    public final <T> T loadTearOff(Class<T> t) {
052        T o = getTearOff(t);
053        if(o==null) {
054            try {
055                o = t.getConstructor(getClass()).newInstance(this);
056                setTearOff(t,o);
057            } catch (InstantiationException e) {
058                throw new InstantiationError(e.getMessage());
059            } catch (IllegalAccessException e) {
060                throw new IllegalAccessError(e.getMessage());
061            } catch (InvocationTargetException e) {
062                Throwable ex = e.getTargetException();
063                if(ex instanceof RuntimeException)
064                    throw (RuntimeException)ex;
065                if(ex instanceof Error)
066                    throw (Error)ex;
067                throw new Error(e);
068            } catch (NoSuchMethodException e) {
069                throw new NoSuchMethodError(e.getMessage());
070            }
071        }
072        return o;
073    }
074
075    public synchronized <T> void setTearOff(Class<T> type, T instance) {
076        Map<Class,Object> m = tearOffs;
077        Map<Class,Object> r = m!=null ? new HashMap<Class, Object>(tearOffs) : new HashMap<Class,Object>();
078        r.put(type,instance);
079        tearOffs = r;
080    }
081}
