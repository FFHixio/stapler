044/**
045 * Objects exported and bound by JavaScript proxies.
046 *
047 * TODO: think about some kind of eviction strategy, beyond the session eviction.
048 * Maybe it's not necessary, I don't know.
049 *
050 * @author Kohsuke Kawaguchi
051 */
052public class BoundObjectTable implements StaplerFallback {
053    public Table getStaplerFallback() {
054        return resolve(false);
055    }
056
057    private Bound bind(Ref ref) {
058        return resolve(true).add(ref);
059    }
060
061    /**
062     * Binds an object temporarily and returns its URL.
063     */
064    public Bound bind(Object o) {
065        return bind(new StrongRef(o));
066    }
067
068    /**
069     * Binds an object temporarily and returns its URL.
070     */
071    public Bound bindWeak(Object o) {
072        return bind(new WeakRef(o));
073    }
074
075    /**
076     * Called from within the request handling of a bound object, to release the object explicitly.
077     */
078    public void releaseMe() {
079        Ancestor eot = Stapler.getCurrentRequest().findAncestor(BoundObjectTable.class);
080        if (eot==null)
081            throw new IllegalStateException("The thread is not handling a request to a abound object");
082        String id = eot.getNextToken(0);
083
084        resolve(false).release(id); // resolve(false) can't fail because we are processing this request now.
085    }
086
087    /**
088     * Obtains a {@link Table} associated with this session.
089     */
090    private Table resolve(boolean createIfNotExist) {
091        HttpSession session = Stapler.getCurrentRequest().getSession(createIfNotExist);
092        if (session==null) return null;
093
094        Table t = (Table) session.getAttribute(Table.class.getName());
095        if (t==null) {
096            if (createIfNotExist)
097                session.setAttribute(Table.class.getName(), t=new Table());
098            else
099                return null;
100        }
101        return t;
102    }
103
104    /**
105     * Explicit call to create the table if one doesn't exist yet.
106     */
107    public Table getTable() {
108        return resolve(true);
109    }
110
111    /**
112     * Per-session table that remembers all the bound instances.
113     */
114    public static class Table {
115        private final Map<String,Ref> entries = new HashMap<String,Ref>();
116        private boolean logging;
117
118        private synchronized Bound add(Ref ref) {
119            final Object target = ref.get();
120            if (target instanceof WithWellKnownURL) {
121                WithWellKnownURL w = (WithWellKnownURL) target;
122                String url = w.getWellKnownUrl();
123                if (!url.startsWith("/")) {
124                    LOGGER.warning("WithWellKnownURL.getWellKnownUrl must start with a slash. But we got " + url + " from "+w);
125                }
126                return new WellKnownObjectHandle(url, w);
127            }
128
129            final String id = UUID.randomUUID().toString();
130            entries.put(id,ref);
131            if (logging)    LOGGER.info(String.format("%s binding %s for %s", toString(), target, id));
132
133            return new Bound() {
134                public void release() {
135                   Table.this.release(id);
136                }
137
138                public String getURL() {
139                    return Stapler.getCurrentRequest().getContextPath()+PREFIX+id;
140                }
141
142                public Object getTarget() {
143                    return target;
144                }
145
146                public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
147                    rsp.sendRedirect2(getURL());
148                }
149            };
150        }
151
152        public Object getDynamic(String id) {
153            return resolve(id);
154        }
155
156        private synchronized Ref release(String id) {
157            return entries.remove(id);
158        }
159
160        private synchronized Object resolve(String id) {
161            Ref e = entries.get(id);
162            if (e==null) {
163                if (logging)    LOGGER.info(toString()+" doesn't have binding for "+id);
164                return null;
165            }
166            Object v = e.get();
167            if (v==null) {
168                if (logging)    LOGGER.warning(toString() + " had binding for " + id + " but it got garbage collected");
169                entries.remove(id); // reference is already garbage collected.
170            }
171            return v;
172        }
173
174        @SuppressFBWarnings(value = "IS2_INCONSISTENT_SYNC", justification = "This usage does not create synchronization problems.")
175        public HttpResponse doEnableLogging() {
176            if (DEBUG_LOGGING) {
177                this.logging = true;
178                return HttpResponses.text("Logging enabled for this session: " + this + "\n");
179            } else {
180                return HttpResponses.forbidden();
181            }
182        }
183    }
184
185    private static final class WellKnownObjectHandle extends Bound {
186        private final String url;
187        private final Object target;
188
189        public WellKnownObjectHandle(String url, Object target) {
190            this.url = url;
191            this.target = target;
192        }
193
194        /**
195         * Objects with well-known URLs cannot be released, as their URL bindings are controlled
196         * implicitly by the application.
197         */
198        public void release() {
199        }
200
201        public String getURL() {
202            return Stapler.getCurrentRequest().getContextPath()+url;
203        }
204
205        @Override
206        public Object getTarget() {
207            return target;
208        }
209
210        public void generateResponse(StaplerRequest req, StaplerResponse rsp, Object node) throws IOException, ServletException {
211            rsp.sendRedirect2(getURL());
212        }
213    }
214
215
216    /**
217     * Reference that resolves to an object.
218     */
219    interface Ref {
220        Object get();
221    }
222    
223    private static class StrongRef implements Ref {
224        private final Object o;
225        StrongRef(Object o) {
226            this.o = o;
227        }
228        public Object get() {
229            return o;
230        }
231    }
232    
233    private static class WeakRef extends WeakReference implements Ref {
234        private WeakRef(Object referent) {
235            super(referent);
236        }
237    }
238
239    public static final String PREFIX = "/$stapler/bound/";

244    @SuppressFBWarnings(value = "MS_SHOULD_BE_FINAL", justification = "Legacy switch.")
245    public static boolean DEBUG_LOGGING = Boolean.getBoolean(BoundObjectTable.class.getName()+".debugLog");
246
247    private static final Logger LOGGER = Logger.getLogger(BoundObjectTable.class.getName());
